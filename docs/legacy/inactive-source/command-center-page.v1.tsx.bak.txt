function formatClock(value: Date): string {
  return value.toLocaleTimeString("en-GB", {
    hour: "2-digit",
    minute: "2-digit",
    hour12: false,
  });
}

function formatMoment(value: string | null | undefined): string {
  if (!value) return "Not yet";

  return new Date(value).toLocaleString("en-GB", {
    day: "2-digit",
    month: "short",
    hour: "2-digit",
    minute: "2-digit",
  });
}

function formatCompactMoney(value: number): string {
  return new Intl.NumberFormat("en", {
    notation: "compact",
    maximumFractionDigits: 1,
  }).format(value);
}

function getStatusTone(character: ConsoleCharacter): string {
  if (character.status === "critical") return "#F87171";
  if (character.status === "at_risk") return "#FB923C";
  if (character.status === "watch") return "#FACC15";
  return "#86CD7E";
}

function getImportanceTone(tier: "anchor" | "core" | "support"): string {
  if (tier === "anchor") return "#F3C567";
  if (tier === "core") return "#86D1FF";
  return "#9CB4E8";
}

function getModeScore(
  character: ConsoleCharacter,
  mode: OpsMode,
  projectFit: number,
): number {
  const hpPct = (character.hp / character.maxHp) * 100;

  if (mode === "hrt") {
    return Math.round(
      hpPct * 0.35 +
        character.fourC.community * 0.2 +
        character.fourC.career * 0.2 +
        character.fourC.cause * 0.15 +
        (100 - character.utilization) * 0.1
    );
  }

  if (mode === "moneyball") {
    return Math.round(
      projectFit * 0.4 +
        character.ica.overall * 0.3 +
        character.form * 6 +
        character.investmentValue * 0.18 -
        character.capacityCost * 0.28 +
        character.demandCount * 5
    );
  }

  return Math.round(
    projectFit * 0.45 +
      character.ica.overall * 0.25 +
      character.form * 6 +
      (character.isPresent ? 12 : 0) +
      (character.isRemote ? -4 : 6)
  );
}

function getModeLabel(mode: OpsMode): string {
  if (mode === "hrt") return "Human signal";
  if (mode === "moneyball") return "Value index";
  return "Board signal";
}

function getActionLabel(character: ConsoleCharacter): string {

}

type DragOrigin = "roster" | "sandbox" | "team" | "coach";

interface DragPayload {
  id: string;
  origin: DragOrigin;
}

function clampPercentage(value: number, min: number, max: number) {
  return Math.min(max, Math.max(min, value));
}

function resolveDropPosition(
  event: DragEvent<HTMLElement>,
  padding = { x: 6, y: 10 },
) {
  const rect = event.currentTarget.getBoundingClientRect();
  const rawX = ((event.clientX - rect.left) / rect.width) * 100;
  const rawY = ((event.clientY - rect.top) / rect.height) * 100;

  return {
    x: clampPercentage(rawX, padding.x, 100 - padding.x),
    y: clampPercentage(rawY, padding.y, 100 - padding.y),
  };
}

function CandidateMarket({
  entries,
  mode,
  demand,
  team,
  project,
  hoveredCandidateId,
  onHover,
  onInspect,
  onDragStart,
  onDragEnd,
  onStage,
  onSnap,
}: {
  entries: Array<{
    character: ConsoleCharacter;
    importance: ReturnType<typeof getCharacterImportance>;
    projectFit: number;
    modeScore: number;
    selected: boolean;
    staged: boolean;
  }>;
  mode: OpsMode;

  team: ConsoleCharacter[];
  project: (typeof TKC_REAL_PROJECTS)[number];
  hoveredCandidateId: string | null;
  onHover: (id: string | null) => void;
  onInspect: (id: string) => void;
  onDragStart: (id: string) => void;
  onDragEnd: () => void;
  onStage: (id: string) => void;
  onSnap: (id: string) => void;
}) {
  return (
    <section className="border border-[#2c1f12] bg-[rgba(15,10,6,0.92)] px-4 py-3">
      <div className="mb-3 flex items-center justify-between">
        <div>
          <div className="type-label text-[#7a6b54]">Candidate Market</div>
          <div className="text-[12px] text-[#e4d9c7]">Pull from the roster, stage in free space, or snap straight into the team.</div>
        </div>
        <div className="text-[10px] text-[#8e7b63]">{entries.length} available</div>
      </div>

      <div className="flex gap-3 overflow-x-auto pb-1">
        {entries.map((entry) => {
          const delta = demand ? previewSnapDelta(team, entry.character, project.id) : [];
          const bestDelta = delta.reduce(
            (best, item) => (item.delta > best.delta ? item : best),
            { key: "technical", label: "Technical", delta: 0, before: 0, after: 0, demand: 0 } as const,
          );

          return (
            <div
              key={entry.character.id}
              onMouseEnter={() => onHover(entry.character.id)}
              onMouseLeave={() => onHover(null)}
            >
              <PoolCard
                entry={entry}
                mode={mode}
                canAdd
                onAdd={() => onSnap(entry.character.id)}
                onInspect={() => onInspect(entry.character.id)}
                onDragStart={() => onDragStart(entry.character.id)}
                onDragEnd={onDragEnd}
                onStage={() => onStage(entry.character.id)}
                badge={
                  bestDelta.delta > 0
                    ? `+${bestDelta.delta}% ${bestDelta.label}`
                    : entry.character.id === hoveredCandidateId
                      ? "Inspect"
                      : undefined
                }
              />
            </div>
          );
        })}
      </div>
    </section>
  );
}

function ScoutConsole({

  onInspect,

              <BarRow label="In-House" value={skillProfile.inHouse} tone="#FB923C" />
            </div>
          )}

          <div className="grid grid-cols-2 gap-2">
            <MetricCard label="ICA" value={String(focus.ica.overall)} tone="#86D1FF" compact />
            <MetricCard label="Form" value={focus.form.toFixed(1)} tone="#F3C567" compact />
            <MetricCard label="HP" value={`${Math.round((focus.hp / focus.maxHp) * 100)}%`} tone={getStatusTone(focus)} compact />
            <MetricCard label="Cost" value={`฿${focus.capacityCost}k`} tone="#FB923C" compact />

                  {snapRead.canSnap ? `Snap score ${snapRead.score}` : "Cannot snap cleanly"}
                </div>
                <div className="space-y-1 text-[10px] text-[#9c8b73]">
                  {snapRead.reasons.length > 0 ? (
                    snapRead.reasons.map((reason) => <div key={reason}>• {reason}</div>)

          </div>
        </>
      ) : (

function FinanceBars({
  series,
}: {
  series: typeof TKC_FINANCIAL_SERIES;
}) {
  const maxRevenue = Math.max(...series.map((point) => point.revenueM));
  const maxProfit = Math.max(...series.map((point) => Math.max(point.grossProfitM, point.netProfitM)));

  return (
    <div className="space-y-3 border border-[#322314] bg-[rgba(12,8,4,0.58)] px-3 py-3">
      {series.map((point) => (
        <div key={point.period} className="space-y-2">
          <div className="flex items-center justify-between text-[10px] text-[#a18d73]">
            <span>{point.period}</span>
            <span>GM {point.grossMarginPct}% · OCF {point.operatingCashflowM}M</span>
          </div>
          <div className="space-y-1">
            <div>
              <div className="mb-1 flex items-center justify-between text-[9px] text-[#7f6c57]">
                <span>Revenue</span>
                <span>{point.revenueM.toFixed(0)}M</span>
              </div>
              <div className="h-2 bg-[#171009]">
                <div className="h-full" style={{ width: `${(point.revenueM / maxRevenue) * 100}%`, background: "#86D1FF" }} />
              </div>
            </div>
            <div>
              <div className="mb-1 flex items-center justify-between text-[9px] text-[#7f6c57]">
                <span>Gross Profit</span>
                <span>{point.grossProfitM.toFixed(0)}M</span>
              </div>
              <div className="h-2 bg-[#171009]">
                <div className="h-full" style={{ width: `${(point.grossProfitM / maxProfit) * 100}%`, background: "#F3C567" }} />
              </div>
            </div>
            <div>
              <div className="mb-1 flex items-center justify-between text-[9px] text-[#7f6c57]">
                <span>Net Profit</span>
                <span>{point.netProfitM.toFixed(0)}M</span>
              </div>
              <div className="h-2 bg-[#171009]">
                <div className="h-full" style={{ width: `${(point.netProfitM / maxProfit) * 100}%`, background: "#86CD7E" }} />
              </div>
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}

function QuarterTape({
  points,
}: {
  points: typeof TKC_QUARTERLY_REVENUE_2025;
}) {
  const maxValue = Math.max(...points.map((point) => point.revenueM));

  return (
    <div className="border border-[#322314] bg-[rgba(12,8,4,0.58)] px-3 py-3">
      <div className="flex items-end gap-2">
        {points.map((point) => (
          <div key={point.period} className="flex flex-1 flex-col items-center gap-2">
            <div className="flex h-26 w-full items-end bg-[#171009] px-1 py-1">
              <div
                className="w-full"
                style={{
                  height: `${Math.max(12, (point.revenueM / maxValue) * 100)}%`,
                  background: point.period === "Q3'25" ? "#F3C567" : "#86D1FF",
                }}
              />
            </div>
            <div className="text-center text-[9px] text-[#8e7b63]">
              <div>{point.period}</div>
              <div>{point.revenueM.toFixed(0)}M</div>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
