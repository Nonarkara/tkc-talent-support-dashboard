function parseNullableNumber(value: unknown): number | null {
  if (typeof value === "number" && Number.isFinite(value)) return value;
  if (typeof value === "string" && value.trim() !== "") {
    const parsed = Number(value);
    return Number.isFinite(parsed) ? parsed : null;
  }
  return null;
}

function createDraftId(prefix: string) {
  return `${prefix}-${Date.now().toString(36)}-${Math.random().toString(36).slice(2, 8)}`;
}

  return { text: "", target: null, current: null, unit: "" };

function normalizeKeyResults(raw: unknown): OkrKeyResult[] {
  if (!Array.isArray(raw)) return [];

  return raw.flatMap((entry) => {
    if (!entry || typeof entry !== "object") return [];

    const row = entry as Record<string, unknown>;
    const keyResult: OkrKeyResult = {
      text: typeof row.text === "string" ? row.text : "",
      target: parseNullableNumber(row.target),
      current: parseNullableNumber(row.current),
      unit: typeof row.unit === "string" ? row.unit : "",
    };

    return keyResult.text || keyResult.target !== null || keyResult.current !== null || keyResult.unit
      ? [keyResult]
      : [];
  });
}

function normalizeOkr(raw: unknown, index: number): OkrRecord {
  const row = raw && typeof raw === "object" ? raw as Record<string, unknown> : {};

  return {
    id: typeof row.id === "string" ? row.id : createDraftId("okr-fallback"),
    cycle: typeof row.cycle === "string" ? row.cycle : CYCLE,
    objective: typeof row.objective === "string" ? row.objective : "",
    key_results: normalizeKeyResults(row.key_results),
    status: typeof row.status === "string" ? row.status : "active",
    sort_order: parseNullableNumber(row.sort_order) ?? index,
    updated_at: typeof row.updated_at === "string" ? row.updated_at : null,
  };
}

function sanitizeKeyResults(keyResults: OkrKeyResult[]) {
  return keyResults
    .map((keyResult) => ({
      text: keyResult.text.trim(),
      current: keyResult.current,
      target: keyResult.target,
      unit: keyResult.unit.trim(),
    }))
    .filter((keyResult) =>
      keyResult.text ||
      keyResult.current !== null ||
      keyResult.target !== null ||
      keyResult.unit
    );
}

function sortSupportActions(actions: SupportActionRecord[]) {
  const statusOrder: Record<SupportActionStatus, number> = {
    in_progress: 0,
    planned: 1,
    done: 2,
    dropped: 3,
  };

  return [...actions].sort((left, right) => {
    const statusGap = statusOrder[left.status] - statusOrder[right.status];
    if (statusGap !== 0) return statusGap;

    const leftTime = left.updated_at ? new Date(left.updated_at).getTime() : 0;
    const rightTime = right.updated_at ? new Date(right.updated_at).getTime() : 0;
    return rightTime - leftTime;
  });
}

function getSupportActionSummary(actions: SupportActionRecord[]) {
  if (actions.length === 0) return "No private action attached yet.";

  const inProgressCount = actions.filter((action) => action.status === "in_progress").length;
  const labels = actions.slice(0, 2).map((action) => action.title);
  const extra = actions.length > 2 ? ` +${actions.length - 2} more` : "";

  return `${actions.length} active action${actions.length === 1 ? "" : "s"}${inProgressCount > 0 ? ` · ${inProgressCount} in progress` : ""} · ${labels.join(" · ")}${extra}`;
}

function getOkrProgress(okr: OkrRecord) {
  if (okr.status === "done") return 100;

  const measurable = okr.key_results.filter(
    (keyResult) => keyResult.target !== null && keyResult.target > 0 && keyResult.current !== null
  );

  if (measurable.length === 0) return 0;

  const total = measurable.reduce((sum, keyResult) => {
    const ratio = (keyResult.current ?? 0) / (keyResult.target ?? 1);
    return sum + Math.max(0, Math.min(100, ratio * 100));
  }, 0);

  return Math.round(total / measurable.length);
}

function getLeadershipReadiness(emp: Employee) {
  const roleBonus =
    emp.role_level === "director" ? 14 :
    emp.role_level === "manager" ? 10 :
    emp.role_level === "senior" ? 6 :
    2;

  const levelScore = Math.min(100, emp.level * 8);
  const tenureScore = Math.min(100, emp.tenure_years * 12);

  const score = Math.round(
    getAttrScore(emp.attr_cha) * 0.28 +
    getAttrScore(emp.attr_wis) * 0.24 +
    getAttrScore(emp.attr_con) * 0.18 +
    getAttrScore(emp.attr_int) * 0.12 +
    levelScore * 0.1 +
    tenureScore * 0.08 +
    roleBonus
  );

  return Math.max(0, Math.min(100, score));
}

function isKpiLagging(kpi: DeptKpi) {
  if (kpi.status === "at_risk" || kpi.status === "off_track") return true;
  if (
    kpi.target_value !== null &&
    kpi.target_value > 0 &&
    kpi.actual_value !== null
  ) {
    return Number(kpi.actual_value) < Number(kpi.target_value);
  }
  return false;
}

function createEmptyPriorityWeightState(): PriorityWeights {
  return { captain: 0, tech: 0, sales: 0, ops: 0, scout: 0, fighter: 0, goofoff: 0 };
}

function sumPriorityWeights(weights: PriorityWeights) {
  return ARCHETYPES.reduce((sum, archetype) => sum + (weights[archetype] ?? 0), 0);
}

function normalizePriorityWeightsTotal(raw: PriorityWeights): PriorityWeights {
  const cleaned = ARCHETYPES.reduce((acc, archetype) => {
    acc[archetype] = Math.max(0, Math.round(raw[archetype] ?? 0));
    return acc;
  }, createEmptyPriorityWeightState());

  const total = sumPriorityWeights(cleaned);
  if (total === TEAM_TOKEN_CAP) return cleaned;
  if (total === 0) return defaultPriorityWeights();

  const scaled = ARCHETYPES.map((archetype) => ({
    archetype,
    exact: (cleaned[archetype] / total) * TEAM_TOKEN_CAP,
  }));
  const base = scaled.reduce((acc, item) => {
    acc[item.archetype] = Math.floor(item.exact);
    return acc;
  }, createEmptyPriorityWeightState());

  let remainder = TEAM_TOKEN_CAP - sumPriorityWeights(base);
  const byFraction = [...scaled].sort((left, right) => {
    const diff = (right.exact - Math.floor(right.exact)) - (left.exact - Math.floor(left.exact));
    return diff !== 0 ? diff : ARCHETYPES.indexOf(left.archetype) - ARCHETYPES.indexOf(right.archetype);
  });

  let cursor = 0;
  while (remainder > 0) {
    base[byFraction[cursor % byFraction.length].archetype] += 1;
    remainder -= 1;
    cursor += 1;
  }

  return base;
}

function createEmptyTeamNeedState(): TeamNeedState {
  return { market: 0, technical: 0, leadership: 0 };
}

function sumTeamNeedTokens(state: TeamNeedState) {
  return TEAM_NEED_ORDER.reduce((sum, key) => sum + state[key], 0);
}

function normalizeTeamNeedState(raw: TeamNeedState): TeamNeedState {
  const cleaned = TEAM_NEED_ORDER.reduce((acc, key) => {
    acc[key] = Math.max(0, Math.round(raw[key] ?? 0));
    return acc;
  }, createEmptyTeamNeedState());

  const total = sumTeamNeedTokens(cleaned);
  if (total === TEAM_TOKEN_CAP) return cleaned;
  if (total === 0) return { market: 3, technical: 5, leadership: 2 };

  const scaled = TEAM_NEED_ORDER.map((key) => ({
    key,
    exact: (cleaned[key] / total) * TEAM_TOKEN_CAP,
  }));
  const base = scaled.reduce((acc, item) => {
    acc[item.key] = Math.floor(item.exact);
    return acc;
  }, createEmptyTeamNeedState());

  let remainder = TEAM_TOKEN_CAP - sumTeamNeedTokens(base);
  const byFraction = [...scaled].sort((left, right) => {
    const diff = (right.exact - Math.floor(right.exact)) - (left.exact - Math.floor(left.exact));
    return diff !== 0 ? diff : TEAM_NEED_ORDER.indexOf(left.key) - TEAM_NEED_ORDER.indexOf(right.key);
  });

  let cursor = 0;
  while (remainder > 0) {
    base[byFraction[cursor % byFraction.length].key] += 1;
    remainder -= 1;
    cursor += 1;
  }

  return base;
}

function deriveSuggestedTeamNeedState(project: Project): TeamNeedState {
  const draft = createEmptyTeamNeedState();

  for (const skill of project.required_skills) {
    const code = skill.trim().toUpperCase();
    if (code === "SALES" || code === "BIZ_DEV") {
      draft.market += 3;
    } else if (code === "PM") {
      draft.leadership += 2;
    } else {
      draft.technical += 3;
    }
  }

  const deptCode = (project.dept_code ?? "").toUpperCase();
  if (deptCode === "SALES" || deptCode === "BIZ_DEV") draft.market += 2;
  if (["DIGITAL", "NET_DEL", "ENTERPRISE", "PUB_SAFETY", "IT"].includes(deptCode)) draft.technical += 2;
  if (project.priority === "critical") draft.leadership += 2;
  if (project.priority === "high") draft.leadership += 1;
  if (project.team_size >= 6) draft.leadership += 1;

  return normalizeTeamNeedState(draft);
}

function deriveSuggestedPriorityWeights(project: Project): PriorityWeights {
  const teamNeed = deriveSuggestedTeamNeedState(project);
  const draft = createEmptyPriorityWeightState();

  draft.captain =
    teamNeed.leadership * 0.46 +
    (project.priority === "critical" ? 0.8 : project.priority === "high" ? 0.4 : 0);
  draft.tech = teamNeed.technical * 0.42;
  draft.sales = teamNeed.market * 0.62;
  draft.ops =
    teamNeed.technical * 0.22 +
    teamNeed.leadership * 0.16 +
    (project.team_size >= 6 ? 0.9 : 0.3);
  draft.scout =
    teamNeed.technical * 0.16 +
    teamNeed.market * 0.16 +
    teamNeed.leadership * 0.18 +
    (project.required_skills.length >= 3 ? 0.8 : 0.2);
  draft.fighter = teamNeed.technical * 0.25;
  draft.goofoff = 0.1; // Baseline wildcard

  for (const skill of project.required_skills) {
    const code = skill.trim().toUpperCase();
    if (code === "PM") {
      draft.captain += 0.8;
      draft.scout += 0.6;
    } else if (code === "SALES" || code === "BIZ_DEV") {
      draft.sales += 1.1;
      draft.scout += 0.3;
    } else {
      draft.tech += 0.55;
      draft.ops += 0.35;
    }
  }

  if ((project.gross_margin_pct ?? 100) < 20) {
    draft.sales += 0.4;
    draft.ops += 0.3;
  }

  return normalizePriorityWeightsTotal(draft);
}

function getEmployeeStateCode(employee: Employee) {
  return employee.employee_code ?? employee.id;
}

function getEmployeeTokenCost(employee: Employee) {
  let cost =
    employee.role_level === "md" || employee.role_level === "deputy_md" ? 5 :
    employee.role_level === "director" ? 4 :
    employee.role_level === "manager" ? 3 :
    employee.role_level === "senior" ? 2 :
    1;

  if (employee.level >= 12) cost += 1;
  else if (employee.level >= 8) cost += 0.5;

  if ((employee.salary_thb ?? 0) >= 120_000) cost += 1;
  else if ((employee.salary_thb ?? 0) >= 80_000) cost += 0.5;

  return Math.max(1, Math.min(5, Math.round(cost)));
}

function normalizeTeamNeedProfile(raw: TeamNeedProfile): TeamNeedProfile {
  const total = raw.market + raw.technical + raw.leadership;
  if (total <= 0) return { market: 34, technical: 33, leadership: 33 };

  return {
    market: Math.round((raw.market / total) * 100),
    technical: Math.round((raw.technical / total) * 100),
    leadership: Math.round((raw.leadership / total) * 100),
  };
}

function getEmployeeNeedProfile(employee: Employee): TeamNeedProfile {
  const attrs = getEmployeeAttributes(employee);
  const deptCode = (employee.dept_code ?? "").toUpperCase();

  let market =
    getAttrScore(attrs.cha) * 0.52 +
    getAttrScore(attrs.dex) * 0.22 +
    getAttrScore(attrs.wis) * 0.16;
  let technical =
    getAttrScore(attrs.int) * 0.44 +
    getAttrScore(attrs.wis) * 0.22 +
    getAttrScore(attrs.dex) * 0.18 +
    getAttrScore(attrs.con) * 0.08;
  let leadership =
    getLeadershipReadiness(employee) * 0.68 +
    getAttrScore(attrs.cha) * 0.14 +
    getAttrScore(attrs.con) * 0.08;

  if (deptCode === "SALES" || deptCode === "BIZ_DEV") market += 22;
  if (deptCode === "HR_ADMIN") leadership += 10;
  if (["DIGITAL", "NET_DEL", "ENTERPRISE", "PUB_SAFETY", "IT"].includes(deptCode)) technical += 22;
  if (employee.role_level === "manager" || employee.role_level === "director" || employee.role_level === "deputy_md" || employee.role_level === "md") {
    leadership += 18;
  }

  return normalizeTeamNeedProfile({ market, technical, leadership });
}

function getPrimaryNeedDimension(profile: TeamNeedProfile): TeamNeedDimension {
  return TEAM_NEED_ORDER.reduce((best, key) => (
    profile[key] > profile[best] ? key : best
  ), "technical" as TeamNeedDimension);
}

function getLaneCoveragePct(targetTokens: number, actualSpend: number) {
  if (targetTokens <= 0) return 100;
  return Math.max(0, Math.min(100, Math.round((actualSpend / targetTokens) * 100)));
}

function getArchetypeCoveragePct(targetTokens: number, actualSpend: number) {
  if (targetTokens <= 0) return actualSpend > 0 ? 120 : 100;
  return Math.max(0, Math.min(160, Math.round((actualSpend / targetTokens) * 100)));
}

function serializeTeamNeedState(state: TeamNeedState) {
  return `${TEAM_STATE_TOKEN_PREFIX}${TEAM_NEED_ORDER.map((key) => `${key}=${state[key]}`).join("|")}`;
}

function parseTeamNeedState(insights: string[] | null | undefined) {
  const raw = insights?.find((item) => item.startsWith(TEAM_STATE_TOKEN_PREFIX));
  if (!raw) return null;

  const next = createEmptyTeamNeedState();
  for (const part of raw.slice(TEAM_STATE_TOKEN_PREFIX.length).split("|")) {
    const [key, value] = part.split("=");
    if (key === "market" || key === "technical" || key === "leadership") {
      next[key] = Math.max(0, Math.min(TEAM_TOKEN_CAP, Number(value) || 0));
    }
  }

  return normalizeTeamNeedState(next);
}

function serializeAllocationState(
  coach: Employee | null,
  coachAllocation: number,
  players: Employee[],
  playerAllocations: Record<string, number>,
) {
  const items: string[] = [];

  if (coach) {
    items.push(`coach:${getEmployeeStateCode(coach)}=${coachAllocation}`);
  }

  for (const player of players) {
    const allocation = playerAllocations[player.id] ?? 100;
    items.push(`${getEmployeeStateCode(player)}=${allocation}`);
  }

  return `${TEAM_STATE_ALLOCATION_PREFIX}${items.join("|")}`;
}

function parseAllocationState(
  insights: string[] | null | undefined,
  employees: Employee[],
) {
  const raw = insights?.find((item) => item.startsWith(TEAM_STATE_ALLOCATION_PREFIX));
  const stateCodeToEmployeeId = new Map<string, string>();

  for (const employee of employees) {
    stateCodeToEmployeeId.set(getEmployeeStateCode(employee), employee.id);
  }

  const playerAllocations: Record<string, number> = {};
  let coachAllocation = 100;

  if (!raw) {
    return { coachAllocation, playerAllocations };
  }

  const entries = raw.slice(TEAM_STATE_ALLOCATION_PREFIX.length).split("|").filter(Boolean);
  for (const entry of entries) {
    const [label, valueRaw] = entry.split("=");
    const allocation = Math.max(25, Math.min(100, Number(valueRaw) || 100));

    if (label.startsWith("coach:")) {
      coachAllocation = allocation;
      continue;
    }

    const employeeId = stateCodeToEmployeeId.get(label);
    if (employeeId) {
      playerAllocations[employeeId] = allocation;
    }
  }

  return { coachAllocation, playerAllocations };
}

function isPersistentTeamStateInsight(value: string) {
  return value.startsWith(TEAM_STATE_TOKEN_PREFIX) || value.startsWith(TEAM_STATE_ALLOCATION_PREFIX);
}

function getPatternLabel(profile: TeamNeedProfile) {
  const ordered = TEAM_NEED_ORDER
    .map((key) => ({ key, value: profile[key] }))
    .sort((left, right) => right.value - left.value);

  if (ordered[0].value - ordered[1].value <= 8) return "Balanced Squad";
  return `${TEAM_NEED_META[ordered[0].key].label}-leaning Squad`;
}

function getArchetypeRanking(weights: Record<Archetype, number>) {
  return ARCHETYPES
    .map((archetype) => ({ archetype, value: weights[archetype] ?? 0 }))
    .sort((left, right) => right.value - left.value);
}

function getProjectStyleLabel(weights: PriorityWeights) {
  const ranked = getArchetypeRanking(weights);
  if (ranked[0].value - ranked[1].value <= 1) return "Balanced Playbook";
  return `${ARCHETYPE_LABEL[ranked[0].archetype]}-led Playbook`;
}

function getSquadIdentity(weights: Record<Archetype, number>) {
  const ranked = getArchetypeRanking(weights);
  if (ranked[0].value <= 0) return "Unformed Squad";
  if (ranked[0].value - ranked[1].value <= 0.8) {
    return `${ARCHETYPE_LABEL[ranked[0].archetype]}-${ARCHETYPE_LABEL[ranked[1].archetype]} Core`;
  }
  return `${ARCHETYPE_LABEL[ranked[0].archetype]} Vanguard`;
}

function getEmployeeAttributes(emp: Employee) {
  return {
    str: emp.attr_str ?? 10,
    int: emp.attr_int ?? 10,
    wis: emp.attr_wis ?? 10,
    cha: emp.attr_cha ?? 10,
    dex: emp.attr_dex ?? 10,
    con: emp.attr_con ?? 10,
  };
}

function getEmployeePerformanceScore(emp: Employee) {
  const attrs = getEmployeeAttributes(emp);
  const attrAverage = (attrs.str + attrs.int + attrs.wis + attrs.cha + attrs.dex + attrs.con) / 6;

  return Math.max(
    0,
    Math.min(100, Math.round(attrAverage * 4.2 + emp.level * 1.8 + Math.min(emp.tenure_years, 8) * 1.6))
  );
}

function employeeCoversSkill(emp: Employee, rawSkill: string) {
  const skill = rawSkill.trim().toUpperCase();
  const dept = (emp.dept_code ?? "").toUpperCase();

  if (skill === dept) return true;
  if (skill === "PM") {
    return ["manager", "director", "deputy_md", "md"].includes(emp.role_level);
  }
  if (skill === "SALES") {
    return dept === "SALES" || dept === "BIZ_DEV";
  }
  if (skill === "BIZ_DEV") {
    return dept === "BIZ_DEV" || dept === "SALES";
  }

  return false;
}

function getCoveredSkills(project: Project, members: Employee[]) {
  return project.required_skills.filter((skill) => members.some((member) => employeeCoversSkill(member, skill)));
}

function getTeamChemistryScore(members: Employee[]) {
  if (members.length < 2) return members.length === 1 ? 55 : 0;
  return calculateChemistry(members.map(getEmployeeAttributes)).overall;
}

function getHierarchyIssuesForMembers(members: Employee[], coach: Employee | null) {
  const levels = members.map((member) => member.role_level);
  const hasDirector = levels.some((level) => ["director", "deputy_md", "md"].includes(level));
  const hasManager = levels.some((level) => level === "manager");
  const hasStaff = levels.some((level) => ["staff", "senior"].includes(level));
  const issues: string[] = [];

  if (!coach) issues.push("No coach assigned");
  if (hasDirector && hasStaff && !hasManager) {
    issues.push("Director-to-staff gap needs a manager bridge");
  }

  return issues;
}

function scoreDraftTeam(
  project: Project,
  coach: Employee | null,
  coachAllocationPct: number,
  players: Employee[],
  playerAllocationPcts: Record<string, number>,
  demandTargets: TeamNeedState,
  priorityWeights: PriorityWeights,
): DraftTeamScore {
  const members = coach ? [coach, ...players] : [...players];
  const coveredSkills = getCoveredSkills(project, members);
  const skillFitPct = project.required_skills.length > 0
    ? Math.round((coveredSkills.length / project.required_skills.length) * 100)
    : 100;
  const missingSkills = project.required_skills.filter((skill) => !coveredSkills.includes(skill));
  const chemistryScore = getTeamChemistryScore(members);
  const hierarchyIssues = getHierarchyIssuesForMembers(members, coach);
  const hierarchyScore = hierarchyIssues.length === 0 ? 100 : Math.max(20, 100 - hierarchyIssues.length * 35);

  const demandCoverage = createEmptyTeamNeedState();
  const archetypeSpend = createEmptyPriorityWeightState();
  let tokenSpend = 0;

  if (coach) {
    const profile = getEmployeeNeedProfile(coach);
    const spend = effectiveCost(getEmployeeTokenCost(coach), coachAllocationPct);
    demandCoverage.market += spend * (profile.market / 100);
    demandCoverage.technical += spend * (profile.technical / 100);
    demandCoverage.leadership += spend * (profile.leadership / 100);
    archetypeSpend[getArchetype(coach)] += spend;
    tokenSpend += spend;
  }

  for (const player of players) {
    const allocationPct = playerAllocationPcts[player.id] ?? 100;
    const profile = getEmployeeNeedProfile(player);
    const spend = effectiveCost(getEmployeeTokenCost(player), allocationPct);
    demandCoverage.market += spend * (profile.market / 100);
    demandCoverage.technical += spend * (profile.technical / 100);
    demandCoverage.leadership += spend * (profile.leadership / 100);
    archetypeSpend[getArchetype(player)] += spend;
    tokenSpend += spend;
  }

  const laneFitPct = Math.round(
    TEAM_NEED_ORDER.reduce((sum, key) => sum + getLaneCoveragePct(demandTargets[key], demandCoverage[key]), 0) /
    TEAM_NEED_ORDER.length
  );
  const archetypeTargets = normalizePriorityWeightsTotal(priorityWeights);
  const roundedArchetypeSpend = ARCHETYPES.reduce((acc, archetype) => {
    acc[archetype] = Math.round(archetypeSpend[archetype] * 10) / 10;
    return acc;
  }, createEmptyPriorityWeightState());
  const archetypeCoveragePct = ARCHETYPES.reduce((acc, archetype) => {
    acc[archetype] = getArchetypeCoveragePct(archetypeTargets[archetype], roundedArchetypeSpend[archetype]);
    return acc;
  }, createEmptyPriorityWeightState());
  const archetypeFitPct = Math.round(
    ARCHETYPES.reduce((sum, archetype) => sum + Math.min(100, archetypeCoveragePct[archetype]), 0) /
    ARCHETYPES.length
  );
  const tokenFitPct = Math.round(laneFitPct * 0.65 + archetypeFitPct * 0.35);
  const fitPct = Math.round(skillFitPct * 0.4 + tokenFitPct * 0.6);

  const overspend = Math.max(0, tokenSpend - TEAM_TOKEN_CAP);
  const budgetScore = Math.max(20, Math.round(100 - overspend * 18));

  const sizeGap = Math.abs(members.length - project.team_size);
  const sizeScore = Math.max(30, 100 - sizeGap * 18);
  const squadIdentity = getSquadIdentity(roundedArchetypeSpend);

  const overallScore = Math.round(
    fitPct * 0.36 +
    chemistryScore * 0.24 +
    hierarchyScore * 0.18 +
    budgetScore * 0.1 +
    sizeScore * 0.12
  );

  const insights: string[] = [];
  const tacticalNotes: string[] = [];
  const archetypeDeficits = ARCHETYPES
    .map((archetype) => ({
      archetype,
      gap: Math.round((archetypeTargets[archetype] - roundedArchetypeSpend[archetype]) * 10) / 10,
    }))
    .sort((left, right) => right.gap - left.gap);
  const archetypeSurplus = ARCHETYPES
    .map((archetype) => ({
      archetype,
      gap: Math.round((roundedArchetypeSpend[archetype] - archetypeTargets[archetype]) * 10) / 10,
    }))
    .sort((left, right) => right.gap - left.gap);

  if (coach) insights.push(`Coach assigned: ${coach.nickname ?? coach.full_name_en ?? coach.full_name_th}`);
  else insights.push("Assign a visible coach before locking the lineup");
  insights.push(`${squadIdentity} · ${tokenSpend.toFixed(1)}/${TEAM_TOKEN_CAP} tokens committed`);
  if (missingSkills.length > 0) insights.push(`Missing coverage: ${missingSkills.join(", ")}`);
  else insights.push("All required skills are covered");
  if (tokenFitPct < 70) {
    const weakestLane = TEAM_NEED_ORDER
      .map((key) => ({ key, score: getLaneCoveragePct(demandTargets[key], demandCoverage[key]) }))
      .sort((left, right) => left.score - right.score)[0];
    insights.push(`Focus gap: ${TEAM_NEED_META[weakestLane.key].label} is under-covered`);
  } else {
    insights.push("Token mix matches the project focus well enough to move");
  }
  if (archetypeFitPct < 72) {
    insights.push(`Project DNA gap: ${ARCHETYPE_LABEL[archetypeDeficits[0].archetype]} is still under-covered`);
  } else {
    insights.push("Archetype budget and actual squad shape are aligned");
  }
  if (chemistryScore >= 75) insights.push("Chemistry looks strong enough to move fast");
  else if (chemistryScore < 50) insights.push("Chemistry is fragile; expect coordination drag");
  if (hierarchyIssues.length > 0) insights.push(...hierarchyIssues);
  if (members.length < project.team_size) insights.push(`Need ${project.team_size - members.length} more member${project.team_size - members.length > 1 ? "s" : ""}`);
  if (overspend > 0) insights.push(`Over cap by ${overspend.toFixed(1)} token${overspend >= 2 ? "s" : ""} · reduce allocation or swap cheaper coverage`);
  if (!coach && archetypeTargets.captain >= 3) tacticalNotes.push("Captain-heavy project needs a visible coach");
  if (archetypeDeficits[0].gap > 0.7) tacticalNotes.push(`Need more ${ARCHETYPE_LABEL[archetypeDeficits[0].archetype]} weight to match the playbook`);
  if (archetypeSurplus[0].gap > 0.9) tacticalNotes.push(`${ARCHETYPE_LABEL[archetypeSurplus[0].archetype]} spend is crowding the cap`);
  if (members.length < project.team_size && tokenSpend <= TEAM_TOKEN_CAP - 1.5) tacticalNotes.push("Cap space remains; add one more specialist before locking");
  if (chemistryScore < 55 && archetypeFitPct >= 75) tacticalNotes.push("Structure is close, but the pairing still needs a better chemistry fit");

  return {
    fitPct,
    skillFitPct,
    tokenFitPct,
    archetypeFitPct,
    chemistryScore,
    hierarchyScore,
    budgetScore,
    sizeScore,
    overallScore: Math.max(0, Math.min(100, overallScore)),
    tokenSpend: Math.round(tokenSpend * 10) / 10,
    tokenCap: TEAM_TOKEN_CAP,
    demandCoverage,
    demandTargets,
    archetypeSpend: roundedArchetypeSpend,
    archetypeTargets,
    archetypeCoveragePct,
    squadIdentity,
    missingSkills,
    hierarchyIssues,
    tacticalNotes: tacticalNotes.slice(0, 3),
    insights,
  };
}

function rankCoachCandidates(
  project: Project,
  employees: Employee[],
  selectedPlayerIds: string[],
  demandTargets: TeamNeedState,
  priorityWeights: PriorityWeights,
) {
  const projectPlaybook = normalizePriorityWeightsTotal(priorityWeights);
  const rankedPlaybook = getArchetypeRanking(projectPlaybook);

  return employees
    .filter((emp) => !selectedPlayerIds.includes(emp.id))
    .map((employee): StaffingCandidate => {
      const readiness = getLeadershipReadiness(employee);
      const tokenCost = getEmployeeTokenCost(employee);
      const profile = getEmployeeNeedProfile(employee);
      const archetype = getArchetype(employee);
      const reasons: string[] = [];
      let score = readiness * 0.5;

      if (employeeCoversSkill(employee, "PM")) {
        score += 22;
        reasons.push("Can act as project bridge");
      }
      if (employee.dept_code === project.dept_code) {
        score += 10;
        reasons.push("Owns the home department context");
      }
      if (project.required_skills.some((skill) => employeeCoversSkill(employee, skill))) {
        score += 10;
        reasons.push("Covers at least one required skill");
      }
      if (["manager", "director", "deputy_md", "md"].includes(employee.role_level)) {
        score += 8;
        reasons.push("Role already carries leadership legitimacy");
      } else {
        score -= 18;
      }

      score += (profile.leadership / 100) * (demandTargets.leadership * 4);
      score += projectPlaybook[archetype] * 4.5;
      if (demandTargets.market >= demandTargets.technical && profile.market >= 34) {
        score += 6;
        reasons.push("Can sell the story, not just manage it");
      }
      if (projectPlaybook[archetype] >= rankedPlaybook[1].value) {
        reasons.push(`Matches ${ARCHETYPE_LABEL[archetype]} playbook`);
      }
      score -= tokenCost * 2;
      reasons.push(`${tokenCost} token${tokenCost > 1 ? "s" : ""}`);

      return {
        employee,
        score: Math.round(Math.max(0, Math.min(100, score))),
        reasons,
      };
    })
    .sort((left, right) => right.score - left.score)
    .slice(0, 8);
}

function rankPlayerCandidates(
  project: Project,
  employees: Employee[],
  coach: Employee | null,
  coachAllocationPct: number,
  players: Employee[],
  playerAllocationPcts: Record<string, number>,
  demandTargets: TeamNeedState,
  priorityWeights: PriorityWeights,
) {
  const selectedIds = new Set([
    ...(coach ? [coach.id] : []),
    ...players.map((player) => player.id),
  ]);

  const currentMembers = coach ? [coach, ...players] : [...players];
  const baseChemistry = getTeamChemistryScore(currentMembers);
  const coveredSkills = new Set(getCoveredSkills(project, currentMembers));
  const currentDraft = scoreDraftTeam(
    project,
    coach,
    coachAllocationPct,
    players,
    playerAllocationPcts,
    demandTargets,
    priorityWeights,
  );
  const currentCoverage = currentDraft.demandCoverage;

  return employees
    .filter((emp) => !selectedIds.has(emp.id))
    .map((employee): StaffingCandidate => {
      const reasons: string[] = [];
      const tokenCost = getEmployeeTokenCost(employee);
      const profile = getEmployeeNeedProfile(employee);
      const archetype = getArchetype(employee);
      let score = getEmployeePerformanceScore(employee) * 0.3;

      const filledSkills = project.required_skills.filter(
        (skill) => !coveredSkills.has(skill) && employeeCoversSkill(employee, skill)
      );
      if (filledSkills.length > 0) {
        score += 28 + filledSkills.length * 6;
        reasons.push(`Closes gap: ${filledSkills.join(", ")}`);
      }

      if (employee.dept_code === project.dept_code) {
        score += 8;
        reasons.push("Close to core delivery context");
      }

      const chemistryAfter = getTeamChemistryScore([...currentMembers, employee]);
      const chemistryDelta = chemistryAfter - baseChemistry;
      score += chemistryDelta * 0.35;
      if (chemistryDelta >= 8) reasons.push("Improves team chemistry");
      if (chemistryDelta <= -5) reasons.push("Could create friction");

      const attrs = getEmployeeAttributes(employee);
      const attributeLift = Math.round((attrs.int + attrs.wis + attrs.con) / 3 * 1.2);
      score += attributeLift * 0.18;

      const laneLift =
        Math.max(0, demandTargets.market - currentCoverage.market) * (profile.market / 100) +
        Math.max(0, demandTargets.technical - currentCoverage.technical) * (profile.technical / 100) +
        Math.max(0, demandTargets.leadership - currentCoverage.leadership) * (profile.leadership / 100);
      score += laneLift * 4.5;
      const archetypeGap = Math.max(
        0,
        currentDraft.archetypeTargets[archetype] - currentDraft.archetypeSpend[archetype],
      );
      const archetypeOverstack = Math.max(
        0,
        currentDraft.archetypeSpend[archetype] - currentDraft.archetypeTargets[archetype],
      );
      score += archetypeGap * 8;
      score -= archetypeOverstack * 3.5;
      if (archetypeGap >= 0.8) reasons.push(`Closes ${ARCHETYPE_LABEL[archetype]} gap`);

      const primary = getPrimaryNeedDimension(profile);
      reasons.push(`${TEAM_NEED_META[primary].label} bias · ${tokenCost} token${tokenCost > 1 ? "s" : ""}`);
      score -= tokenCost * 1.8;

      return {
        employee,
        score: Math.round(Math.max(0, Math.min(100, score))),
        reasons,
      };
    })
    .sort((left, right) => right.score - left.score)
    .slice(0, 12);
}

// ─── Evaluation Spider Panel ──────────────────────────────────────────────────

function EvalSpider({ empId, activeRater }: { empId: string; activeRater: RaterType }) {
  const [evals, setEvals] = useState<Evaluation[]>([]);
  const [flashKey, setFlashKey] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    fetch(`/api/db/evaluations?employee_id=${empId}&cycle=${CYCLE}`)
      .then((r) => r.json())
      .then((d) => { if (!cancelled) setEvals(d.evaluations ?? []); });
    return () => { cancelled = true; };
  }, [empId]);

  const datasets: SpiderDataset[] = useMemo(() => {
    const raters: RaterType[] = ["self", "manager", "hr"];
    return raters.map((rt) => {
      const values: Record<string, number> = {};
      for (const d of DIMENSIONS) {
        const row = evals.find((e) => e.dimension_key === d.key && e.rater_type === rt);
        values[d.key] = row ? Number(row.score) : 0;
      }
      return {
        key: rt,
        color: RATER_COLOR[rt],
        style: rt === "self" ? "dashed" : "solid",
        values,
      };
    });
  }, [evals]);

  function setLocal(axisKey: string, newValue: number) {
    setEvals((prev) => {
      const filtered = prev.filter((e) => !(e.dimension_key === axisKey && e.rater_type === activeRater));
      return [...filtered, { dimension_key: axisKey, rater_type: activeRater, score: newValue }];
    });
  }

  async function saveScore(axisKey: string, newValue: number) {
    setLocal(axisKey, newValue);
    try {
      await fetch("/api/db/evaluations", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          employee_id: empId, dimension_key: axisKey,
          rater_type: activeRater, cycle: CYCLE, score: newValue,
        }),
      });
      setFlashKey(axisKey);
      setTimeout(() => setFlashKey((k) => (k === axisKey ? null : k)), 900);
    } catch {}
  }

  // Compute gap chips
  const gaps = DIMENSIONS.map((d) => {
    const s = datasets.find((x) => x.key === "self")!.values[d.key];
    const m = datasets.find((x) => x.key === "manager")!.values[d.key];
    return { key: d.key, label: d.label, gap: Math.abs(s - m) };
  }).filter((g) => g.gap >= 15).sort((a, b) => b.gap - a.gap);

  return (
    <div>
      <div style={{ display: "flex", justifyContent: "center", padding: "8px 0" }}>
        <SpiderWeb
          axes={SPIDER_AXES}
          datasets={datasets}
          size={260}
          activeDatasetKey={activeRater}
          onDrag={setLocal}
          onDragEnd={saveScore}
        />
      </div>

      {/* Legend */}
      <div style={{ display: "flex", justifyContent: "center", gap: 14, marginTop: 2 }}>
        {(["self", "manager", "hr"] as RaterType[]).map((r) => (
          <span key={r} style={{
            display: "flex", alignItems: "center", gap: 5,
            fontFamily: "'JetBrains Mono', monospace", fontSize: 9,
            color: r === activeRater ? RATER_COLOR[r] : "var(--text-muted)",
            fontWeight: r === activeRater ? 700 : 400,
          }}>
            <span style={{ width: 8, height: 8, background: RATER_COLOR[r], display: "inline-block" }} />
            {RATER_LABEL[r]}
          </span>
        ))}
      </div>

      {/* Gap chips */}
      {gaps.length > 0 && (
        <div style={{ marginTop: 14, padding: "10px 12px", border: "1px solid var(--border-subtle)", background: "rgba(196,77,63,0.06)" }}>
          <div style={{ fontSize: 9, color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase", marginBottom: 6 }}>
            Calibration gaps · self vs manager
          </div>
          <div style={{ display: "flex", flexWrap: "wrap", gap: 6 }}>
            {gaps.map((g) => (
              <span key={g.key} style={{
                fontFamily: "'JetBrains Mono', monospace", fontSize: 9,
                background: "rgba(196,77,63,0.16)", color: "#E38477",
                padding: "2px 6px",
              }}>{g.label} · {g.gap}</span>
            ))}
          </div>
        </div>
      )}

      {flashKey && (
        <div style={{ fontSize: 9, textAlign: "center", marginTop: 8, color: "var(--accent-green)", fontFamily: "'JetBrains Mono', monospace" }}>
          ✓ saved · {flashKey}
        </div>
      )}
    </div>
  );
}

function QuestLogPanel({ empId }: { empId: string }) {
  const [okrs, setOkrs] = useState<OkrRecord[]>([]);
  const [loading, setLoading] = useState(true);
  const [message, setMessage] = useState<string | null>(null);
  const [savingId, setSavingId] = useState<string | null>(null);
  const [deletingId, setDeletingId] = useState<string | null>(null);
  const [dbReady, setDbReady] = useState(true);

  useEffect(() => {
    let cancelled = false;

    setLoading(true);
    setMessage(null);

    fetch(`/api/db/okrs?employee_id=${empId}&cycle=${CYCLE}`)
      .then(async (response) => {
        const data = await response.json();
        if (cancelled) return;

        setDbReady(response.status !== 503);
        if (!response.ok && response.status !== 503) {
          setMessage(data.error ?? "Could not load quest log.");
        }

        const rows: unknown[] = Array.isArray(data.okrs) ? data.okrs : [];
        setOkrs(
          rows
            .map((row, index) => normalizeOkr(row, index))
            .sort((left, right) => left.sort_order - right.sort_order)
        );
        setLoading(false);
      })
      .catch(() => {
        if (cancelled) return;
        setMessage("Could not load quest log.");
        setLoading(false);
      });

    return () => {
      cancelled = true;
    };
  }, [empId]);

  const summary = useMemo(() => {
    const active = okrs.filter((okr) => okr.status === "active").length;
    const completed = okrs.filter((okr) => okr.status === "done").length;
    const avgProgress = okrs.length
      ? Math.round(okrs.reduce((sum, okr) => sum + getOkrProgress(okr), 0) / okrs.length)
      : 0;

    return { active, completed, avgProgress };
  }, [okrs]);

  function updateOkr(okrId: string, updater: (okr: OkrRecord) => OkrRecord) {
    setOkrs((current) => current.map((okr) => (okr.id === okrId ? updater(okr) : okr)));
  }

  function addObjective() {
    setOkrs((current) => {
      const nextSortOrder = current.reduce((max, okr) => Math.max(max, okr.sort_order), -1) + 1;
      return [...current, createDraftOkr(nextSortOrder)];
    });
  }

  function addKeyResult(okrId: string) {
    updateOkr(okrId, (okr) => ({
      ...okr,
      key_results: [...okr.key_results, createEmptyKeyResult()],
    }));
  }

  function removeKeyResult(okrId: string, index: number) {
    updateOkr(okrId, (okr) => ({
      ...okr,
      key_results: okr.key_results.filter((_, currentIndex) => currentIndex !== index),
    }));
  }

  function updateKeyResultField(
    okrId: string,
    index: number,
    field: keyof OkrKeyResult,
    value: string | number | null
  ) {
    updateOkr(okrId, (okr) => ({
      ...okr,
      key_results: okr.key_results.map((keyResult, currentIndex) =>
        currentIndex === index ? { ...keyResult, [field]: value } : keyResult
      ),
    }));
  }

  async function saveObjective(okr: OkrRecord) {
    const objective = okr.objective.trim();
    if (!objective) {
      setMessage("Quest title is required before saving.");
      return;
    }

    const keyResults = sanitizeKeyResults(okr.key_results);
    const draft = isDraftOkr(okr.id);

    setSavingId(okr.id);
    setMessage(null);

    try {
      const response = await fetch("/api/db/okrs", {
        method: draft ? "POST" : "PATCH",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(
          draft
            ? {
                employee_id: empId,
                cycle: CYCLE,
                objective,
                key_results: keyResults,
                sort_order: okr.sort_order,
              }
            : {
                id: okr.id,
                objective,
                key_results: keyResults,
                status: okr.status,
                sort_order: okr.sort_order,
              }
        ),
      });
      const data = await response.json();

      if (!response.ok) {
        setMessage(data.error ?? "Could not save quest.");
        return;
      }

      const nextTimestamp = new Date().toISOString();
      if (draft && typeof data.id === "string") {
        setOkrs((current) => current.map((item) => (
          item.id === okr.id
            ? { ...item, id: data.id, objective, key_results: keyResults, updated_at: nextTimestamp }
            : item
        )));
      } else {
        setOkrs((current) => current.map((item) => (
          item.id === okr.id
            ? { ...item, objective, key_results: keyResults, updated_at: nextTimestamp }
            : item
        )));
      }

      setMessage(draft ? "Quest created for this cycle." : "Quest updated.");
    } catch {
      setMessage("Could not save quest.");
    } finally {
      setSavingId(null);
    }
  }

  async function deleteObjective(okrId: string) {
    if (isDraftOkr(okrId)) {
      setOkrs((current) => current.filter((okr) => okr.id !== okrId));
      return;
    }

    setDeletingId(okrId);
    setMessage(null);

    try {
      const response = await fetch(`/api/db/okrs?id=${encodeURIComponent(okrId)}`, {
        method: "DELETE",
      });
      const data = await response.json();

      if (!response.ok) {
        setMessage(data.error ?? "Could not delete quest.");
        return;
      }

      setOkrs((current) => current.filter((okr) => okr.id !== okrId));
      setMessage("Quest removed.");
    } catch {
      setMessage("Could not delete quest.");
    } finally {
      setDeletingId(null);
    }
  }

  return (
    <div style={{ padding: "16px 20px", display: "flex", flexDirection: "column", gap: 14 }}>
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", gap: 12 }}>
        <div>
          <div style={{ fontSize: 9, color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase" }}>
            Quest log · cycle {CYCLE}
          </div>
          <div style={{ display: "flex", flexWrap: "wrap", gap: 8, marginTop: 8, fontFamily: "'JetBrains Mono', monospace", fontSize: 10 }}>
            <span style={{ color: "var(--text-secondary)" }}>{okrs.length} objectives</span>
            <span style={{ color: "#D4A843" }}>{summary.active} active</span>
            <span style={{ color: "#5B8C4A" }}>{summary.completed} done</span>
            <span style={{ color: "var(--text-muted)" }}>{summary.avgProgress}% avg progress</span>
          </div>
        </div>
        <button
          onClick={addObjective}
          disabled={!dbReady}
          style={{
            padding: "7px 10px",
            border: "1px solid var(--accent-gold)",
            background: "rgba(212,168,67,0.12)",
            color: "var(--accent-gold)",
            cursor: dbReady ? "pointer" : "not-allowed",
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: "10px",
            letterSpacing: "0.08em",
          }}
        >
          + Objective
        </button>
      </div>

      {message && (
        <div style={{
          padding: "10px 12px",
          border: "1px solid var(--border-subtle)",
          background: "rgba(91,155,181,0.08)",
          color: "var(--text-secondary)",
          fontSize: "10px",
          fontFamily: "'JetBrains Mono', monospace",
        }}>
          {message}
        </div>
      )}

      {loading && (
        <div style={{ color: "var(--text-muted)", fontSize: "11px", fontFamily: "'JetBrains Mono', monospace" }}>
          Loading quest log...
        </div>
      )}

      {!loading && !dbReady && (
        <div style={{
          padding: "12px 14px",
          border: "1px solid rgba(196,77,63,0.28)",
          background: "rgba(196,77,63,0.08)",
          color: "#E3A199",
          fontSize: "10px",
          lineHeight: 1.6,
          fontFamily: "'JetBrains Mono', monospace",
        }}>
          Database is not configured yet, so the quest log cannot persist objectives. Once Neon is connected, this panel will become live.
        </div>
      )}

      {!loading && dbReady && okrs.length === 0 && (
        <div style={{
          padding: "16px 14px",
          border: "1px dashed var(--border-subtle)",
          color: "var(--text-muted)",
          fontSize: "11px",
          lineHeight: 1.6,
        }}>
          No objectives for this cycle yet. Add the first quest to turn evaluations into an execution plan.
        </div>
      )}

      {!loading && dbReady && okrs.map((okr) => {
        const progress = getOkrProgress(okr);
        const tone = getOkrTone(okr.status);
        const busy = savingId === okr.id || deletingId === okr.id;

        return (
          <div key={okr.id} style={{
            border: `1px solid ${tone}44`,
            borderTop: `3px solid ${tone}`,
            background: "var(--bg-elevated)",
            padding: "14px",
          }}>
            <div style={{ display: "grid", gridTemplateColumns: "1fr 110px", gap: 10 }}>
              <div>
                <label style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", display: "block", marginBottom: 6 }}>
                  Objective
                </label>
                <input

                  Progress

                Key results

                    border: "1px solid var(--border-subtle)",

                    padding: "10px",

                      <span style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "9px", color: "var(--text-secondary)" }}>
                        KR {index + 1}

                          border: "none",
                          background: "transparent",
                          color: "#C44D3F",
                          cursor: "pointer",
                          fontFamily: "'JetBrains Mono', monospace",
                          fontSize: "10px",
                        }}
                      >
                        Remove
                      </button>
                    </div>

                    <input
                      type="text"

                      onChange={(event) => updateKeyResultField(okr.id, index, "text", event.target.value)}
                      placeholder="e.g. 8/10 departments submit weekly check-ins"
                      style={{
                        width: "100%",

                          color: "var(--text-primary)",
                          fontSize: "11px",
                          padding: "7px 10px",
                          fontFamily: "'JetBrains Mono', monospace",
                          outline: "none",

                      />
                      <input

                        onChange={(event) => updateKeyResultField(okr.id, index, "target", parseNullableNumber(event.target.value))}
                        placeholder="Target"
                        style={{
                          background: "var(--bg-surface)",
                          border: "1px solid var(--border-subtle)",

                          padding: "7px 10px",
                          fontFamily: "'JetBrains Mono', monospace",
                          outline: "none",
                        }}

                        type="text"

                        onChange={(event) => updateKeyResultField(okr.id, index, "unit", event.target.value)}
                        placeholder="Unit"
                        style={{
                          background: "var(--bg-surface)",
                          border: "1px solid var(--border-subtle)",
                          color: "var(--text-primary)",
                          fontSize: "11px",
                          padding: "7px 10px",
                          fontFamily: "'JetBrains Mono', monospace",
                          outline: "none",
                        }}
                      />
                    </div>

              </div>
            </div>

            <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginTop: 12 }}>
              <button
                onClick={() => addKeyResult(okr.id)}
                style={{
                  border: "1px solid var(--border-subtle)",
                  background: "transparent",
                  color: "var(--text-secondary)",
                  cursor: "pointer",
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: "10px",
                  padding: "6px 8px",
                }}

                + Key result
              </button>

              <div style={{ display: "flex", gap: 8 }}>
                <button
                  onClick={() => void deleteObjective(okr.id)}
                  disabled={busy}
                  style={{
                    border: "1px solid rgba(196,77,63,0.38)",
                    background: "transparent",

                    cursor: busy ? "not-allowed" : "pointer",
                    fontFamily: "'JetBrains Mono', monospace",

                    padding: "6px 8px",

                >

                </button>

                  onClick={() => void saveObjective(okr)}
                  disabled={busy}
                  style={{
                    border: "none",
                    background: tone,
                    color: "#1a1209",
                    cursor: busy ? "not-allowed" : "pointer",
                    fontFamily: "'JetBrains Mono', monospace",
                    fontSize: "10px",
                    fontWeight: 700,
                    padding: "6px 10px",
                    letterSpacing: "0.08em",
                  }}
                >
                  {savingId === okr.id ? "Saving..." : isDraftOkr(okr.id) ? "Create quest" : "Save quest"}

              </div>
            </div>
          </div>
        );
      })}
    </div>
  );
}

function SupportPlanPanel({
  emp,
  employees,
}: {
  emp: Employee;
  employees: Employee[];
}) {
  const playbook = useMemo(() => getSupportPlaybook(emp), [emp]);
  const signal = getRiskSignal(emp);
  const readiness = getLeadershipReadiness(emp);
  const [actions, setActions] = useState<SupportActionRecord[]>([]);
  const [loading, setLoading] = useState(true);
  const [message, setMessage] = useState<string | null>(null);
  const [dbReady, setDbReady] = useState(true);
  const [savingId, setSavingId] = useState<string | null>(null);
  const [creatingLabel, setCreatingLabel] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;

    setLoading(true);
    setMessage(null);

    fetch(`/api/db/support-actions?employee_id=${emp.id}&cycle=${CYCLE}`)
      .then(async (response) => {
        const data = await response.json();
        if (cancelled) return;

        setDbReady(response.status !== 503);
        if (!response.ok && response.status !== 503) {
          setMessage(data.error ?? "Could not load support workflow.");
        }

        const rows: unknown[] = Array.isArray(data.support_actions) ? data.support_actions : [];
        setActions(sortSupportActions(rows.map(normalizeSupportAction)));
        setLoading(false);
      })
      .catch(() => {
        if (cancelled) return;
        setMessage("Could not load support workflow.");
        setLoading(false);
      });

    return () => {
      cancelled = true;
    };
  }, [emp.id]);

  const summary = useMemo(() => {
    const open = actions.filter(isOpenSupportAction).length;
    const inProgress = actions.filter((action) => action.status === "in_progress").length;
    const done = actions.filter((action) => action.status === "done").length;
    return { open, inProgress, done };
  }, [actions]);

  const ownerOptions = useMemo(() => {
    return [...employees].sort((left, right) => {
      const leftName = left.nickname ?? left.full_name_en ?? left.full_name_th;
      const rightName = right.nickname ?? right.full_name_en ?? right.full_name_th;
      return leftName.localeCompare(rightName);
    });
  }, [employees]);

  const ownerById = useMemo(
    () => new Map(ownerOptions.map((owner) => [owner.id, owner])),
    [ownerOptions],
  );

  const trackedActionLabels = useMemo(() => {
    return new Set(
      actions
        .filter(isOpenSupportAction)
        .map((action) => action.title.toLowerCase()),
    );
  }, [actions]);

  function updateAction(
    actionId: string,
    updater: (action: SupportActionRecord) => SupportActionRecord,
  ) {
    setActions((current) => sortSupportActions(
      current.map((action) => (action.id === actionId ? updater(action) : action)),
    ));
  }

  async function createAction(template: SupportActionTemplate) {
    setCreatingLabel(template.label);
    setMessage(null);

    try {
      const response = await fetch("/api/db/support-actions", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          employee_id: emp.id,
          cycle: CYCLE,
          action_type: template.type,
          title: template.label,
          note: template.note,
        }),
      });
      const data = await response.json();

      if (!response.ok) {
        setMessage(data.error ?? "Could not create support action.");
        return;
      }

      const nextAction = data.support_action ? normalizeSupportAction(data.support_action) : null;
      if (nextAction) {
        setActions((current) => sortSupportActions([nextAction, ...current.filter((action) => action.id !== nextAction.id)]));
      }
      setMessage(`${template.label} added to the private support workflow.`);
    } catch {
      setMessage("Could not create support action.");
    } finally {
      setCreatingLabel(null);
    }
  }

  async function saveAction(action: SupportActionRecord) {
    const title = action.title.trim();
    if (!title) {
      setMessage("Action title cannot be empty.");
      return;
    }

    setSavingId(action.id);
    setMessage(null);

    try {
      const response = await fetch("/api/db/support-actions", {
        method: "PATCH",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          id: action.id,
          title,
          note: action.note,
          status: action.status,
          owner_employee_id: action.owner_employee_id,
        }),
      });
      const data = await response.json();

      if (!response.ok) {
        setMessage(data.error ?? "Could not save support action.");
        return;
      }

      const nextAction = data.support_action ? normalizeSupportAction(data.support_action) : null;
      if (nextAction) {
        setActions((current) => sortSupportActions(
          current.map((currentAction) => (
            currentAction.id === nextAction.id ? nextAction : currentAction
          )),
        ));
      }
      setMessage(`${title} saved.`);
    } catch {
      setMessage("Could not save support action.");
    } finally {
      setSavingId(null);
    }
  }

  return (
    <div style={{ padding: "16px 20px", display: "flex", flexDirection: "column", gap: 14 }}>
      <div style={{
        border: `1px solid ${getSupportTone(signal)}44`,
        borderTop: `3px solid ${getSupportTone(signal)}`,
        background: "var(--bg-elevated)",
        padding: "14px",
      }}>
        <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase" }}>
          Support posture
        </div>
        <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "18px", fontWeight: 700, color: "var(--text-primary)", marginTop: 8 }}>
          {playbook.posture}
        </div>
        <div style={{ fontSize: "11px", color: "var(--text-secondary)", lineHeight: 1.7, marginTop: 8 }}>
          {playbook.summary}
        </div>

        <div style={{ display: "flex", gap: 16, marginTop: 14 }}>
          <div>
            <div style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase" }}>Signal</div>
            <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "12px", color: getSupportTone(signal), marginTop: 4 }}>
              {RISK_LABEL[signal]}
            </div>
          </div>
          <div>
            <div style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase" }}>Leadership Readiness</div>
            <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "12px", color: readiness >= 70 ? "#5B8C4A" : "#D4A843", marginTop: 4 }}>
              {readiness}
            </div>
          </div>
        </div>
      </div>

      <div style={{ display: "flex", flexWrap: "wrap", gap: 8, fontFamily: "'JetBrains Mono', monospace", fontSize: "10px" }}>
        <span style={{ color: "var(--text-secondary)" }}>{actions.length} actions</span>
        <span style={{ color: "#D4A843" }}>{summary.open} open</span>
        <span style={{ color: "#5B89B5" }}>{summary.inProgress} active</span>
        <span style={{ color: "#5B8C4A" }}>{summary.done} done</span>
      </div>

      {message && (
        <div style={{
          padding: "10px 12px",
          border: "1px solid var(--border-subtle)",
          background: "rgba(91,155,181,0.08)",
          color: "var(--text-secondary)",
          fontSize: "10px",
          fontFamily: "'JetBrains Mono', monospace",
        }}>
          {message}
        </div>
      )}

      {!loading && !dbReady && (
        <div style={{
          padding: "12px 14px",
          border: "1px solid rgba(196,77,63,0.28)",
          background: "rgba(196,77,63,0.08)",
          color: "#E3A199",
          fontSize: "10px",
          lineHeight: 1.6,
          fontFamily: "'JetBrains Mono', monospace",
        }}>
          Database is not configured yet, so private support actions cannot persist. Once Neon is connected, this panel becomes a real HR workflow instead of draft guidance only.
        </div>
      )}

      <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 12 }}>
        <section style={{ border: "1px solid var(--border-subtle)", background: "var(--bg-elevated)", padding: "14px" }}>
          <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase", marginBottom: 10 }}>
            Private checks
          </div>
          <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
            {playbook.privateChecks.map((check) => (
              <div key={check} style={{ fontSize: "11px", color: "var(--text-secondary)", lineHeight: 1.6 }}>
                {check}
              </div>
            ))}
          </div>
        </section>

        <section style={{ border: "1px solid var(--border-subtle)", background: "var(--bg-elevated)", padding: "14px" }}>
          <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase", marginBottom: 10 }}>
            Recommended interventions
          </div>
          <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
            {playbook.actions.map((action) => (
              <button
                key={action.label}
                onClick={() => void createAction(action)}
                disabled={!dbReady || trackedActionLabels.has(action.label.toLowerCase()) || creatingLabel === action.label}
                style={{
                  border: `1px solid ${action.tone}44`,
                  background: `${action.tone}14`,
                  color: "var(--text-primary)",
                  textAlign: "left",
                  cursor: dbReady ? "pointer" : "not-allowed",
                  padding: "10px 12px",
                  opacity: !dbReady || trackedActionLabels.has(action.label.toLowerCase()) ? 0.72 : 1,
                }}
              >
                <div style={{ display: "flex", justifyContent: "space-between", gap: 8, alignItems: "center" }}>
                  <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "10px", fontWeight: 700, color: action.tone }}>
                    {action.label}
                  </div>
                  <div style={{ fontSize: "8px", color: trackedActionLabels.has(action.label.toLowerCase()) ? "#5B8C4A" : "var(--text-muted)", letterSpacing: "0.08em", textTransform: "uppercase" }}>
                    {trackedActionLabels.has(action.label.toLowerCase()) ? "Tracked" : creatingLabel === action.label ? "Saving..." : "Add"}
                  </div>
                </div>
                <div style={{ fontSize: "10px", color: "var(--text-secondary)", marginTop: 6, lineHeight: 1.6 }}>
                  {action.note}
                </div>
              </button>
            ))}
          </div>
        </section>
      </div>

      <section style={{ border: "1px solid var(--border-subtle)", background: "var(--bg-elevated)", padding: "14px" }}>
        <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", gap: 10, marginBottom: 10 }}>
          <div>
            <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase" }}>
              Persistent workflow
            </div>
            <div style={{ fontSize: "11px", color: "var(--text-secondary)", marginTop: 4 }}>
              Private HR actions with owner and status. This is the operating layer behind the support doctrine.
            </div>
          </div>
        </div>

        {loading && (
          <div style={{ color: "var(--text-muted)", fontSize: "11px", fontFamily: "'JetBrains Mono', monospace" }}>
            Loading support workflow...
          </div>
        )}

        {!loading && dbReady && actions.length === 0 && (
          <div style={{
            padding: "14px 12px",
            border: "1px dashed var(--border-subtle)",
            color: "var(--text-muted)",
            fontSize: "10px",
            lineHeight: 1.6,
          }}>
            No private actions tracked yet. Use the recommendations above to turn support posture into real follow-through.
          </div>
        )}

        {!loading && dbReady && actions.length > 0 && (
          <div style={{ display: "flex", flexDirection: "column", gap: 10 }}>
            {actions.map((action) => {
              const typeMeta = SUPPORT_ACTION_TYPE_META[action.action_type];
              const statusTone = SUPPORT_ACTION_STATUS_TONE[action.status];
              const busy = savingId === action.id;

              return (
                <div key={action.id} style={{
                  border: `1px solid ${typeMeta.tone}44`,
                  borderTop: `3px solid ${typeMeta.tone}`,
                  background: "rgba(0,0,0,0.08)",
                  padding: "12px",
                }}>
                  <div style={{ display: "flex", justifyContent: "space-between", gap: 12, alignItems: "flex-start" }}>
                    <div>
                      <div style={{ display: "flex", flexWrap: "wrap", gap: 6, alignItems: "center" }}>
                        <span style={{
                          fontFamily: "'JetBrains Mono', monospace",
                          fontSize: "9px",
                          color: typeMeta.tone,
                          letterSpacing: "0.08em",
                          textTransform: "uppercase",
                        }}>
                          {typeMeta.label}
                        </span>
                        <span style={{
                          fontFamily: "'JetBrains Mono', monospace",
                          fontSize: "9px",
                          color: statusTone,
                          letterSpacing: "0.08em",
                          textTransform: "uppercase",
                        }}>
                          {SUPPORT_ACTION_STATUS_LABEL[action.status]}
                        </span>
                      </div>
                      <div style={{ fontSize: "10px", color: "var(--text-muted)", marginTop: 4 }}>
                        Owner: {getSupportOwnerLabel(action)}
                        {action.updated_at
                          ? ` · ${new Date(action.updated_at).toLocaleDateString("en-GB", { day: "2-digit", month: "short" })}`
                          : ""}
                      </div>
                    </div>
                    <button
                      onClick={() => void saveAction(action)}
                      disabled={busy}
                      style={{
                        border: "none",
                        background: typeMeta.tone,
                        color: "#1a1209",
                        cursor: busy ? "not-allowed" : "pointer",
                        fontFamily: "'JetBrains Mono', monospace",
                        fontSize: "10px",
                        fontWeight: 700,
                        padding: "6px 10px",
                        letterSpacing: "0.08em",
                      }}
                    >
                      {busy ? "Saving..." : "Save"}
                    </button>
                  </div>

                  <div style={{ display: "grid", gridTemplateColumns: "1.2fr 0.8fr 0.8fr", gap: 8, marginTop: 10 }}>
                    <div>
                      <label style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", display: "block", marginBottom: 6 }}>
                        Action
                      </label>
                      <input
                        type="text"
                        value={action.title}
                        onChange={(event) => updateAction(action.id, (current) => ({ ...current, title: event.target.value }))}
                        style={{
                          width: "100%",
                          background: "var(--bg-surface)",
                          border: "1px solid var(--border-subtle)",
                          color: "var(--text-primary)",
                          fontSize: "11px",

                          outline: "none",
                        }}
                      />
                    </div>
                    <div>
                      <label style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", display: "block", marginBottom: 6 }}>

                          const owner = ownerById.get(event.target.value);
                          updateAction(action.id, (current) => ({
                            ...current,
                            owner_employee_id: event.target.value || null,
                            owner_nickname: owner?.nickname ?? null,
                            owner_full_name_en: owner?.full_name_en ?? null,
                            owner_full_name_th: owner?.full_name_th ?? null,
                          }));
                        }}
                        style={{
                          width: "100%",
                          background: "var(--bg-surface)",
                          border: "1px solid var(--border-subtle)",
                          color: "var(--text-primary)",
                          fontSize: "11px",
                          padding: "7px 10px",
                          fontFamily: "'JetBrains Mono', monospace",
                          outline: "none",
                        }}
                      >
                        <option value="">Unassigned</option>
                        {ownerOptions.map((owner) => (
                          <option key={owner.id} value={owner.id}>
                            {owner.nickname ?? owner.full_name_en ?? owner.full_name_th}
                          </option>
                        ))}
                      </select>
                    </div>
                    <div>
                      <label style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", display: "block", marginBottom: 6 }}>
                        Status
                      </label>
                      <select
                        value={action.status}
                        onChange={(event) => updateAction(action.id, (current) => ({
                          ...current,
                          status: event.target.value as SupportActionStatus,
                        }))}
                        style={{
                          width: "100%",
                          background: "var(--bg-surface)",
                          border: "1px solid var(--border-subtle)",

function ProjectStaffingDrawer({
  project,
  team,
  employees,
  onClose,
  onSaved,
  onProjectUpdated,
}: {
  project: Project;
  team?: TeamComposition;
  employees: Employee[];
  onClose: () => void;
  onSaved: (projectCode: string) => void;
  onProjectUpdated?: (projectCode: string, priorityWeights: PriorityWeights) => void;
}) {
  const employeeById = useMemo(
    () => new Map(employees.map((employee) => [employee.id, employee])),
    [employees]
  );
  const suggestedNeeds = useMemo(
    () => deriveSuggestedTeamNeedState(project),
    [project]
  );
  const suggestedPriorityWeights = useMemo(
    () => deriveSuggestedPriorityWeights(project),
    [project]
  );

  const [coachId, setCoachId] = useState<string | null>(team?.coach_id ?? null);
  const [playerIds, setPlayerIds] = useState<string[]>(team?.player_ids ?? []);
  const [formation, setFormation] = useState(team?.formation ?? "442");
  const [teamNeeds, setTeamNeeds] = useState<TeamNeedState>(suggestedNeeds);
  const [priorityWeights, setPriorityWeights] = useState<PriorityWeights>(
    normalizePriorityWeightsTotal(
      project.priority_weights ? normalizeWeights(project.priority_weights) : suggestedPriorityWeights,
    ),
  );
  const [coachAllocationPct, setCoachAllocationPct] = useState(100);
  const [playerAllocationPcts, setPlayerAllocationPcts] = useState<Record<string, number>>({});
  const [message, setMessage] = useState<string | null>(null);
  const [saving, setSaving] = useState(false);

  useEffect(() => {
    const parsedNeeds = parseTeamNeedState(team?.insights) ?? suggestedNeeds;
    const legacyAllocations = parseAllocationState(team?.insights, employees);
    // Prefer the canonical jsonb column; fall back to the legacy insights
    // serialization for rows written before migration 008.
    const jsonbAllocations = team?.allocation_pcts ?? null;
    const playerAllocations = jsonbAllocations && Object.keys(jsonbAllocations).length > 0
      ? Object.fromEntries(
          Object.entries(jsonbAllocations).filter(
            ([, pct]) => pct === 50 || pct === 100 || pct === 25 || pct === 75,
          ),
        )
      : legacyAllocations.playerAllocations;

    setCoachId(team?.coach_id ?? null);
    setPlayerIds(team?.player_ids ?? []);
    setFormation(team?.formation ?? "442");
    setTeamNeeds(parsedNeeds);
    setPriorityWeights(
      normalizePriorityWeightsTotal(
        project.priority_weights ? normalizeWeights(project.priority_weights) : suggestedPriorityWeights,
      ),
    );
    setCoachAllocationPct(legacyAllocations.coachAllocation);
    setPlayerAllocationPcts(playerAllocations);
    setMessage(null);
  }, [employees, project.code, project.priority_weights, suggestedNeeds, suggestedPriorityWeights, team?.coach_id, team?.formation, team?.insights, team?.player_ids, team?.allocation_pcts]);

  const coach = coachId ? employeeById.get(coachId) ?? null : null;
  const players = playerIds
    .map((playerId) => employeeById.get(playerId))
    .filter((employee): employee is Employee => Boolean(employee));

  const draftScore = useMemo(
    () => scoreDraftTeam(project, coach, coachAllocationPct, players, playerAllocationPcts, teamNeeds, priorityWeights),
    [project, coach, coachAllocationPct, players, playerAllocationPcts, priorityWeights, teamNeeds]
  );

  const coachCandidates = useMemo(
    () => rankCoachCandidates(project, employees, playerIds, teamNeeds, priorityWeights),
    [project, employees, playerIds, priorityWeights, teamNeeds]
  );

  const playerCandidates = useMemo(
    () => rankPlayerCandidates(project, employees, coach, coachAllocationPct, players, playerAllocationPcts, teamNeeds, priorityWeights),
    [project, employees, coach, coachAllocationPct, players, playerAllocationPcts, priorityWeights, teamNeeds]
  );
  const priorityRanking = useMemo(
    () => getArchetypeRanking(priorityWeights),
    [priorityWeights]
  );

  function togglePlayer(playerId: string) {
    setPlayerIds((current) => {
      const next = current.includes(playerId)
        ? current.filter((value) => value !== playerId)
        : [...current, playerId];

      return next;
    });
    setPlayerAllocationPcts((current) => {
      if (playerId in current) {
        const next = { ...current };
        delete next[playerId];
        return next;
      }
      return { ...current, [playerId]: 100 };
    });
  }

  function adjustTeamNeed(dimension: TeamNeedDimension, delta: 1 | -1) {
    setTeamNeeds((current) => {
      const total = sumTeamNeedTokens(current);

      if (delta > 0 && total >= TEAM_TOKEN_CAP) return current;
      if (delta < 0 && current[dimension] <= 0) return current;

      return { ...current, [dimension]: current[dimension] + delta };
    });
  }

  function adjustPriorityWeight(archetype: Archetype, delta: 1 | -1) {
    setPriorityWeights((current) => normalizePriorityWeightsTotal(
      rebalanceWeights(current, archetype, current[archetype] + delta),
    ));
  }

  function setPlayerAllocation(playerId: string, nextValue: number) {
    setPlayerAllocationPcts((current) => ({
      ...current,
      [playerId]: Math.max(25, Math.min(100, nextValue)),
    }));
  }

  async function handleSave() {
    setSaving(true);
    setMessage(null);

    const persistedInsights = [
      ...draftScore.insights.filter((item) => !isPersistentTeamStateInsight(item)),
      serializeTeamNeedState(teamNeeds),
      serializeAllocationState(coach, coachAllocationPct, players, playerAllocationPcts),
    ];

    try {
      const normalizedPriorityWeights = normalizePriorityWeightsTotal(priorityWeights);
      let prioritySaved = false;
      let priorityWarning: string | null = null;

      try {
        const priorityResponse = await fetch("/api/db/projects", {
          method: "PATCH",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            project_code: project.code,
            priority_weights: normalizedPriorityWeights,
          }),
        });
        const priorityData = await priorityResponse.json().catch(() => ({}));

        if (priorityResponse.ok) {
          prioritySaved = true;
          onProjectUpdated?.(project.code, normalizedPriorityWeights);
        } else {
          priorityWarning = priorityData.error ?? "Project playbook could not be saved.";
        }
      } catch {
        priorityWarning = "Project playbook could not be saved.";
      }

      const persistedAllocations: Record<string, 25 | 50 | 75 | 100> = {};
      for (const player of players) {
        const raw = playerAllocationPcts[player.id] ?? 100;
        const normalized =
          raw <= 25 ? 25 :
          raw <= 50 ? 50 :
          raw <= 75 ? 75 :
          100;
        persistedAllocations[player.id] = normalized;
      }

      const response = await fetch("/api/db/teams", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          project_code: project.code,
          coach_id: coach?.id ?? null,
          coach_code: coach?.employee_code ?? null,
          player_ids: players.map((player) => player.id),
          player_codes: players.flatMap((player) => player.employee_code ? [player.employee_code] : []),
          formation,
          selector_mode: "hr",
          fit_pct: draftScore.fitPct,
          chemistry_score: draftScore.chemistryScore,
          overall_score: draftScore.overallScore,
          insights: persistedInsights,
          allocation_pcts: persistedAllocations,
        }),
      });
      const data = await response.json();

      if (!response.ok) {
        setMessage(
          prioritySaved
            ? `Project playbook saved, but lineup could not be saved: ${data.error ?? "Unknown error."}`
            : data.error ?? priorityWarning ?? "Could not save lineup.",
        );
        return;
      }

      setMessage(prioritySaved ? "Lineup and project playbook saved." : `Lineup saved. ${priorityWarning ?? ""}`.trim());
      onSaved(project.code);
    } catch {
      setMessage("Could not save lineup.");
    } finally {
      setSaving(false);
    }
  }

  return (
    <div
      style={{ position: "fixed", inset: 0, zIndex: 60, background: "rgba(0,0,0,0.62)", backdropFilter: "blur(2px)" }}
      onClick={(event) => { if (event.target === event.currentTarget) onClose(); }}
    >
      <div style={{
        position: "absolute",
        right: 0,
        top: 0,
        bottom: 0,
        width: "760px",
        background: "var(--bg-surface)",
        borderLeft: "1px solid var(--border-subtle)",
        display: "flex",
        flexDirection: "column",
      }}>
        <div style={{ padding: "20px 24px 16px", borderBottom: "1px solid var(--border-subtle)", background: "var(--bg-elevated)" }}>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", gap: 16 }}>
            <div>
              <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase" }}>
                Staffing Planner
              </div>
              <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "20px", fontWeight: 700, color: "var(--text-primary)", marginTop: 8 }}>
                {project.code} · {project.name}
              </div>
              <div style={{ fontSize: "11px", color: "var(--text-secondary)", marginTop: 6 }}>
                {project.client_name ?? "Internal"} · {project.dept_code ?? "—"} · priority {project.priority}
              </div>
            </div>

            <button onClick={onClose} style={{
              background: "transparent",
              border: "none",
              color: "var(--text-muted)",
              cursor: "pointer",
              fontSize: "18px",
              lineHeight: 1,
            }}>
              ×
            </button>
          </div>

          <div style={{ display: "grid", gridTemplateColumns: "repeat(5, minmax(0, 1fr))", gap: 8, marginTop: 14 }}>
            {[
              { label: "Fit", value: `${draftScore.fitPct}%`, tone: "#D4A843" },
              { label: "Chemistry", value: `${draftScore.chemistryScore}%`, tone: "#5B89B5" },
              { label: "Overall", value: `${draftScore.overallScore}%`, tone: draftScore.overallScore >= 75 ? "#5B8C4A" : "#C4943F" },
              {
                label: "Spend",
                value: `${draftScore.tokenSpend.toFixed(1)}/${draftScore.tokenCap}`,
                tone: draftScore.tokenSpend > draftScore.tokenCap ? "#C44D3F" : "#5B8C4A",
              },
              { label: "Headcount", value: `${(coach ? 1 : 0) + players.length}/${project.team_size}`, tone: "#C44D3F" },
            ].map((item) => (
              <div key={item.label} style={{ border: `1px solid ${item.tone}44`, padding: "10px 12px", background: "rgba(0,0,0,0.08)" }}>
                <div style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase" }}>{item.label}</div>
                <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "18px", fontWeight: 700, color: item.tone, marginTop: 6 }}>{item.value}</div>
              </div>
            ))}
          </div>
        </div>

        <div style={{ flex: 1, minHeight: 0, overflowY: "auto", padding: "16px 20px", display: "grid", gridTemplateColumns: "0.92fr 1.08fr", gap: 14 }}>
          <div style={{ display: "flex", flexDirection: "column", gap: 14 }}>
            <section style={{ border: "1px solid var(--border-subtle)", background: "var(--bg-elevated)", padding: "14px" }}>
              <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", gap: 12, marginBottom: 10 }}>
                <div>
                  <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase" }}>
                    Project Focus
                  </div>
                  <div style={{ fontSize: "11px", color: "var(--text-secondary)", marginTop: 4 }}>
                    Give the director 10 hidden tokens and decide what this quest truly needs.
                  </div>
                </div>
                <button
                  onClick={() => setTeamNeeds(suggestedNeeds)}
                  style={{
                    border: "1px solid var(--border-subtle)",
                    background: "transparent",
                    color: "var(--text-secondary)",
                    cursor: "pointer",
                    fontFamily: "'JetBrains Mono', monospace",
                    fontSize: "9px",
                    padding: "6px 8px",
                  }}
                >
                  Reset to suggested

                Saved team scores will feed the staffing history instead of living only as a workshop artifact.
              </div>
              <button onClick={() => void handleSave()} disabled={saving} style={{
                border: "none",
                background: "var(--accent-gold)",
                color: "#1a1209",
                cursor: saving ? "not-allowed" : "pointer",
                fontFamily: "'JetBrains Mono', monospace",
                fontSize: "11px",
                fontWeight: 700,
                padding: "10px 14px",
                letterSpacing: "0.08em",
              }}>
                {saving ? "Saving..." : "Save lineup"}
              </button>
            </div>
          </div>
        </div>

    </div>

function CockpitScreen({
  employees,
  projects,
  teams,
  supportActions,
  kpis,
  onSelect,
  onOpenScreen,
  onFocusDept,
  onOpenProject,
}: {
  employees: Employee[];
  projects: Project[];
  teams: TeamComposition[];
  supportActions: SupportActionRecord[];
  kpis: DeptKpi[];
  onSelect: (emp: Employee) => void;
  onOpenScreen: (screen: Screen) => void;
  onFocusDept: (deptCode: string) => void;
  onOpenProject: (projectCode: string) => void;
}) {
  const teamByProject = useMemo(() => {
    const map = new Map<string, TeamComposition>();
    for (const team of teams) {
      if (!map.has(team.project_code)) {
        map.set(team.project_code, team);
      }
    }
    return map;
  }, [teams]);

  const employeeById = useMemo(
    () => new Map(employees.map((e) => [e.id, e])),
    [employees],
  );

  const supportActionsByEmployee = useMemo(() => {
    const map = new Map<string, SupportActionRecord[]>();
    for (const action of supportActions) {
      const current = map.get(action.employee_id) ?? [];
      current.push(action);
      map.set(action.employee_id, current);
    }
    return map;
  }, [supportActions]);

  const kpisByDept = useMemo(() => {
    const map: Record<string, DeptKpi[]> = {};
    for (const kpi of kpis) {
      if (!map[kpi.dept_code]) map[kpi.dept_code] = [];
      map[kpi.dept_code].push(kpi);
    }
    return map;
  }, [kpis]);

  const supportQueue = useMemo(() => (
    employees
      .filter((emp) => {
        const signal = getRiskSignal(emp);
        return signal === "risk" || signal === "watch";
      })
      .map((emp) => {
        const signal = getRiskSignal(emp);
        const openActions = getOpenSupportActionsForEmployee(supportActionsByEmployee, emp.id);

        return {
          employee: emp,
          signal,
          openActions,
          priority:
            (signal === "risk" ? 40 : 20) +
            (HIGH_RISK_DEPTS.has(emp.dept_code ?? "") ? 10 : 0) +
            (emp.tenure_years === 0 ? 6 : 0) +
            (emp.role_level === "manager" ? 4 : 0) +
            (openActions.length === 0 ? 18 : 0) +
            (openActions.some((action) => action.status === "in_progress") ? -6 : 0),
        };
      })
      .sort((left, right) => right.priority - left.priority || left.employee.tenure_years - right.employee.tenure_years)
      .slice(0, 6)
  ), [employees, supportActionsByEmployee]);

  const formationQueue = useMemo(() => (
    projects
      .filter((project) => project.priority === "critical" || project.priority === "high")
      .map((project) => {
        const team = teamByProject.get(project.code);
        const assigned = getTeamMemberCount(team);
        const gap = Math.max(0, project.team_size - assigned);
        return {
          project,
          team,
          assigned,
          gap,
          action: getProjectAction(project, team),
        };
      })
      .sort((left, right) => {
        const leftUrgency = (left.team ? 0 : 20) + left.gap * 4 + getPriorityRank(left.project.priority) * -3;
        const rightUrgency = (right.team ? 0 : 20) + right.gap * 4 + getPriorityRank(right.project.priority) * -3;
        return rightUrgency - leftUrgency;
      })
      .slice(0, 6)
  ), [projects, teamByProject]);

  const successionBench = useMemo(() => (
    employees
      .filter((emp) => emp.role_level !== "md" && emp.role_level !== "deputy_md")
      .map((emp) => ({
        employee: emp,
        readiness: getLeadershipReadiness(emp),
        track: getLeadershipTrack(emp),
        reason: getLeadershipReason(emp),
      }))
      .sort((left, right) => right.readiness - left.readiness)
      .slice(0, 6)
  ), [employees]);

  const departmentHeat = useMemo(() => {
    const deptCodes = Array.from(new Set([
      ...employees.map((emp) => emp.dept_code).filter(Boolean) as string[],
      ...Object.keys(kpisByDept),
    ]));

    return deptCodes
      .map((deptCode) => {
        const members = employees.filter((emp) => emp.dept_code === deptCode);
        const deptKpis = kpisByDept[deptCode] ?? [];
        const riskCount = members.filter((emp) => {
          const signal = getRiskSignal(emp);
          return signal === "risk" || signal === "watch";
        }).length;
        const anchorCount = members.filter((emp) => getRiskSignal(emp) === "anchor").length;
        const laggingCount = deptKpis.filter(isKpiLagging).length;
        const strain = riskCount * 18 + laggingCount * 10 + Math.max(0, 2 - anchorCount) * 4;

        const archetypeMix: Record<Archetype, number> = {
          captain: 0, tech: 0, sales: 0, ops: 0, scout: 0, fighter: 0, goofoff: 0,
        };
        for (const member of members) archetypeMix[getArchetype(member)] += 1;

        return {
          deptCode,
          deptName: members[0]?.dept_name_en ?? deptCode,
          riskCount,
          anchorCount,
          laggingCount,
          memberCount: members.length,
          strain,
          archetypeMix,
        };
      })
      .sort((left, right) => right.strain - left.strain)
      .slice(0, 6);
  }, [employees, kpisByDept]);

  const archetypeCensus = useMemo(() => {
    const counts: Record<Archetype, number> = { 
      captain: 0, tech: 0, sales: 0, ops: 0, scout: 0, fighter: 0, goofoff: 0 
    };
    for (const emp of employees) counts[getArchetype(emp)] += 1;
    const total = employees.length || 1;
    return ARCHETYPES.map((archetype) => ({
      archetype,
      count: counts[archetype],
      pct: Math.round((counts[archetype] / total) * 100),
    }));
  }, [employees]);
  const archetypeCensusTotal = employees.length;
  const archetypeDominant = archetypeCensus.reduce(
    (best, entry) => (entry.count > best.count ? entry : best),
    archetypeCensus[0] ?? { archetype: "ops" as Archetype, count: 0, pct: 0 },
  );
  const archetypeScarcest = archetypeCensus.reduce(
    (worst, entry) => (entry.count < worst.count ? entry : worst),
    archetypeCensus[0] ?? { archetype: "ops" as Archetype, count: 0, pct: 0 },
  );

  const privateQueueCount = supportQueue.length;
  const activeSupportActionCount = useMemo(
    () => supportActions.filter(isOpenSupportAction).length,
    [supportActions],
  );
  const coveredSupportCount = useMemo(
    () => supportQueue.filter((entry) => entry.openActions.length > 0).length,
    [supportQueue],
  );
  const uncoveredSupportCount = privateQueueCount - coveredSupportCount;
  const successionReadyCount = successionBench.filter((entry) => entry.readiness >= 70).length;
  const projectGapCount = formationQueue.filter((item) => !item.team || item.gap > 0).length;
  const heatDeptCount = departmentHeat.filter((dept) => dept.strain > 0).length;

  if (employees.length === 0 && projects.length === 0) {
    return (
      <div style={{ padding: "24px", maxWidth: "980px", margin: "0 auto" }}>
        <div style={{
          background: "var(--bg-elevated)",
          border: "1px solid rgba(196,77,63,0.28)",
          borderTop: "3px solid #C44D3F",
          padding: "18px 20px",
        }}>
          <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase" }}>
            Cockpit Waiting For Live Data
          </div>
          <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "20px", fontWeight: 700, color: "var(--text-primary)", marginTop: 8 }}>
            No employee or quest data has loaded yet.
          </div>
          <div style={{ fontSize: "12px", color: "var(--text-secondary)", lineHeight: 1.7, marginTop: 10 }}>
            This usually means the database is not configured yet, migrations have not been applied, or the live tables are still empty. Once the roster and project data arrive, this page will become the manager and HR operating cockpit.
          </div>
        </div>
      </div>
    );
  }

  return (
    <div style={{
      padding: "48px 32px 96px",
      maxWidth: 1080,
      margin: "0 auto",
      display: "flex",
      flexDirection: "column",
    }}>

      {/* ── Masthead ───────────────────────────────────────── */}
      <div style={{
        display: "flex",
        justifyContent: "space-between",
        alignItems: "flex-end",
        paddingBottom: 20,
        borderBottom: "1px solid var(--border-subtle)",
        gap: 24,
        flexWrap: "wrap",
      }}>
        <div>
          <div style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: 9,
            letterSpacing: "0.14em",
            textTransform: "uppercase",
            color: "var(--text-muted)",
          }}>
            Manager + HR Cockpit
          </div>
          <div style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: 18,
            fontWeight: 400,
            color: "var(--text-primary)",
            marginTop: 10,
            letterSpacing: "-0.005em",
          }}>
            Public myth, private instrumentation.
          </div>
        </div>
        <div style={{
          fontFamily: "'JetBrains Mono', monospace",
          fontSize: 10,
          letterSpacing: "0.14em",
          textTransform: "uppercase",
          color: "var(--text-muted)",
          paddingBottom: 4,
        }}>
          2026 · Q2
        </div>
      </div>

      {/* ── 01 · Roster DNA ────────────────────────────────── */}
      <section style={{ marginTop: 40 }}>
        <div style={{
          display: "flex",
          justifyContent: "space-between",
          alignItems: "baseline",
          marginBottom: 28,
        }}>
          <div style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: 10,
            letterSpacing: "0.14em",
            textTransform: "uppercase",
            color: "var(--text-muted)",
          }}>
            01 · Roster DNA
          </div>
          <div style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: 9,
            letterSpacing: "0.12em",
            textTransform: "uppercase",
            color: "var(--text-muted)",
          }}>
            {archetypeCensusTotal} People
          </div>
        </div>

        <div style={{
          display: "grid",
          gridTemplateColumns: "repeat(5, minmax(0, 1fr))",
          gap: 32,
          marginBottom: 24,
        }}>
          {archetypeCensus.map(({ archetype, count, pct }) => {

            return (

                display: "flex",
                flexDirection: "column",
                gap: 10,

                <div style={{
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: 36,
                  fontWeight: 300,
                  color: tone,
                  lineHeight: 1,

          lineHeight: 1.65,
          maxWidth: 640,

          {archetypeCensusTotal > 0 ? (
            <>
              TKC leans{" "}
              <span style={{
                color: ARCHETYPE_COLOR[archetypeDominant.archetype],
                fontFamily: "'JetBrains Mono', monospace",
              }}>
                {ARCHETYPE_LABEL[archetypeDominant.archetype]}
              </span>
              {" · thin on "}
              <span style={{
                color: ARCHETYPE_COLOR[archetypeScarcest.archetype],
                fontFamily: "'JetBrains Mono', monospace",

                {ARCHETYPE_LABEL[archetypeScarcest.archetype]}

              . Direct hire, redeploy, or coach to rebalance.

          )}
        </div>
      </section>

      {/* ── 02 · Attention ─────────────────────────────────── */}
      <section style={{
        marginTop: 48,
        paddingTop: 40,
        borderTop: "1px solid var(--border-subtle)",
      }}>
        <div style={{
          display: "flex",
          justifyContent: "space-between",
          alignItems: "baseline",
          marginBottom: 24,
        }}>
          <div style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: 10,
            letterSpacing: "0.14em",
            textTransform: "uppercase",
            color: "var(--text-muted)",

            02 · Attention
          </div>
          <div style={{
            fontFamily: "'JetBrains Mono', monospace",
            fontSize: 9,
            letterSpacing: "0.12em",
            textTransform: "uppercase",

          }}>

          </div>
        </div>

        <div>
          {supportQueue.map(({ employee, signal, openActions }, idx) => {
            const archetype = getArchetype(employee);
            const tone = ARCHETYPE_COLOR[archetype];
            const tokenCost = getEmployeeTokenCost(employee);
            const isLast = idx === supportQueue.length - 1 && formationQueue.length === 0;
            return (

                borderLeft: `2px solid ${tone}`,
                borderBottom: isLast ? "none" : "1px solid var(--border-subtle)",
                background: "transparent",
                textAlign: "left",
                cursor: "pointer",
                color: "inherit",
              }}>
                <span style={{
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: 11,
                  color: tone,
                  letterSpacing: "0.04em",
                }}>
                  ⚡{tokenCost}
                </span>
                <div>
                  <div style={{
                    fontFamily: "'JetBrains Mono', monospace",
                    fontSize: 12,
                    fontWeight: 600,
                    color: "var(--text-primary)",
                  }}>
                    {employee.nickname ?? employee.full_name_en?.split(" ")[0] ?? "—"}
                  </div>
                  <div style={{
                    fontFamily: "'JetBrains Mono', monospace",
                    fontSize: 9,
                    color: "var(--text-muted)",
                    marginTop: 3,
                    letterSpacing: "0.06em",
                  }}>
                    {(ROLE_LABEL[employee.role_level] ?? employee.role_level).toUpperCase()} · {employee.dept_code ?? "—"}

                  fontSize: 10,
                  color: getSupportTone(signal),
                  letterSpacing: "0.06em",
                  textTransform: "uppercase",
                }}>
                  {RISK_LABEL[signal]}
                </div>

                  fontSize: 12,

            <div style={{
              padding: "24px 18px",

        marginTop: 48,
        paddingTop: 40,

            fontSize: 10,
            letterSpacing: "0.14em",

            <div style={{
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: 9,

// ─── Signals Screen ───────────────────────────────────────────────────────────

function SignalsScreen({
  employees,
  supportActions,
  onSelect,
}: {
  employees: Employee[];
  supportActions: SupportActionRecord[];
  onSelect: (emp: Employee) => void;
}) {
  const supportActionsByEmployee = useMemo(() => {
    const map = new Map<string, SupportActionRecord[]>();
    for (const action of supportActions) {
      const current = map.get(action.employee_id) ?? [];
      current.push(action);
      map.set(action.employee_id, current);
    }
    return map;
  }, [supportActions]);

  const atRisk = employees.filter((e) => {
    const r = getRiskSignal(e);
    return r === "risk" || r === "watch";
  }).sort((a, b) => {
    const order = { risk: 0, watch: 1 };
    const ra = order[getRiskSignal(a) as "risk" | "watch"] ?? 2;
    const rb = order[getRiskSignal(b) as "risk" | "watch"] ?? 2;
    return ra - rb || a.tenure_years - b.tenure_years;
  });

  const anchors = employees.filter((e) => getRiskSignal(e) === "anchor")
    .sort((a, b) => b.tenure_years - a.tenure_years).slice(0, 10);

  const riskByArchetype = useMemo(() => {
    const out: Record<Archetype, { risk: number; total: number }> = {
      captain: { risk: 0, total: 0 },
      tech:    { risk: 0, total: 0 },
      sales:   { risk: 0, total: 0 },
      ops:     { risk: 0, total: 0 },
      scout:   { risk: 0, total: 0 },
      fighter: { risk: 0, total: 0 },
      goofoff: { risk: 0, total: 0 },
    };
    for (const e of employees) {
      const archetype = getArchetype(e);
      out[archetype].total += 1;
      const signal = getRiskSignal(e);
      if (signal === "risk" || signal === "watch") {
        out[archetype].risk += 1;
      }
    }
    return out;
  }, [employees]);

  const coveredCount = atRisk.filter((employee) => (
    getOpenSupportActionsForEmployee(supportActionsByEmployee, employee.id).length > 0
  )).length;
  const activeActionCount = supportActions.filter(isOpenSupportAction).length;

  return (
    <div style={{ padding: "24px", maxWidth: "900px", margin: "0 auto" }}>
      <div style={{ marginBottom: 24 }}>
        <div style={{ fontSize: "10px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", marginBottom: 4 }}>
          Flight Risk · {atRisk.length} people
        </div>
        <div style={{ fontSize: "11px", color: "var(--text-secondary)" }}>
          New joiners and high-attrition departments (Procurement, Accounting, Digital) with under 2 years tenure.
        </div>
        <div style={{ display: "flex", gap: 10, flexWrap: "wrap", marginTop: 10, fontFamily: "'JetBrains Mono', monospace", fontSize: "9px" }}>
          <span style={{ color: "#C44D3F" }}>{atRisk.length - coveredCount} uncovered</span>
          <span style={{ color: "#5B8C4A" }}>{coveredCount} covered</span>
          <span style={{ color: "#5B89B5" }}>{activeActionCount} active support actions</span>
        </div>

        <div style={{ marginTop: 14, borderTop: "1px solid var(--border-subtle)", paddingTop: 10 }}>
          <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", marginBottom: 8 }}>
            Risk by Archetype · scarcity-weighted
          </div>
          <div style={{ display: "grid", gridTemplateColumns: "repeat(5, minmax(0, 1fr))", gap: 8 }}>
            {ARCHETYPES.map((archetype) => {
              const bucket = riskByArchetype[archetype];
              const ratio = bucket.total > 0 ? bucket.risk / bucket.total : 0;
              const alarming = bucket.total > 0 && ratio >= 0.35;
              const tone = ARCHETYPE_COLOR[archetype];
              const fillPct = Math.min(100, Math.round(ratio * 100));
              return (
                <div
                  key={archetype}
                  title={`${bucket.risk} of ${bucket.total} ${ARCHETYPE_LABEL[archetype]} people flagged`}
                  style={{
                    borderLeft: `2px solid ${alarming ? "#C44D3F" : tone}`,
                    background: `${tone}10`,
                    padding: "6px 8px 7px",
                    display: "flex",
                    flexDirection: "column",
                    gap: 4,
                  }}
                >
                  <div style={{ display: "flex", justifyContent: "space-between", alignItems: "baseline", gap: 4 }}>
                    <span style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "9px", fontWeight: 700, color: tone, letterSpacing: "0.06em" }}>
                      {ARCHETYPE_LABEL[archetype].slice(0, 3).toUpperCase()}
                    </span>
                    <span style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "8px", color: alarming ? "#C44D3F" : "var(--text-muted)" }}>
                      {bucket.risk}/{bucket.total}
                    </span>
                  </div>
                  <div style={{ height: 4, background: `${tone}22`, overflow: "hidden" }}>
                    <div style={{
                      height: "100%",
                      width: `${fillPct}%`,
                      background: alarming ? "#C44D3F" : tone,
                      transition: "width 0.3s",
                    }} />
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      </div>

      {atRisk.length === 0 ? (
        <div style={{ color: "var(--text-muted)", fontSize: "12px" }}>No signals detected.</div>
      ) : (
        <div style={{ display: "flex", flexDirection: "column", gap: 6, marginBottom: 32 }}>
          {atRisk.map((emp) => {
            const risk = getRiskSignal(emp);
            const openActions = getOpenSupportActionsForEmployee(supportActionsByEmployee, emp.id);
            const archetype = getArchetype(emp);
            const archetypeTone = ARCHETYPE_COLOR[archetype];
            const tokenCost = getEmployeeTokenCost(emp);
            return (
              <button key={emp.id} onClick={() => onSelect(emp)} style={{
                display: "grid", gridTemplateColumns: "120px 120px 80px 1fr", alignItems: "center", gap: 12,
                background: "var(--bg-elevated)", border: `1px solid ${RISK_COLOR[risk]}44`,
                borderLeft: `3px solid ${archetypeTone}`,
                padding: "10px 14px", textAlign: "left", cursor: "pointer", transition: "border-color 0.15s",
              }}>
                <div>
                  <div style={{ display: "flex", alignItems: "center", gap: 8 }}>
                    <span style={{ width: 8, height: 8, borderRadius: "50%", background: RISK_COLOR[risk], flexShrink: 0, display: "inline-block" }} />
                    <span style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "12px", fontWeight: 700, color: "var(--text-primary)" }}>
                      {emp.nickname ?? emp.full_name_en?.split(" ")[0] ?? "—"}
                    </span>
                  </div>
                  <div style={{ fontSize: "9px", color: RISK_COLOR[risk], marginTop: 5 }}>
                    {RISK_LABEL[risk]}
                  </div>
                </div>
                <div style={{ display: "flex", flexDirection: "column", gap: 3 }}>
                  <div style={{ display: "flex", alignItems: "center", gap: 6 }}>
                    <span style={{
                      fontFamily: "'JetBrains Mono', monospace",
                      fontSize: "9px",
                      fontWeight: 700,
                      color: archetypeTone,
                      letterSpacing: "0.06em",
                    }}>
                      {ARCHETYPE_LABEL[archetype].toUpperCase()}
                    </span>
                    <span style={{
                      fontFamily: "'JetBrains Mono', monospace",
                      fontSize: "9px",
                      color: "var(--text-muted)",
                    }}>
                      ⚡{tokenCost}
                    </span>
                  </div>
                  <span style={{ fontSize: "9px", color: "var(--text-muted)" }}>
                    {emp.dept_name_en ?? emp.dept_code}
                  </span>
                </div>
                <div style={{ fontSize: "10px", color: "var(--text-muted)" }}>
                  {emp.tenure_years === 0 ? "< 1 year" : `${emp.tenure_years} yr`}
                </div>
                <div style={{
                  fontFamily: "'JetBrains Mono', monospace",
                  fontSize: "9px",
                  color: openActions.length > 0 ? "#5B8C4A" : "#E38477",
                  lineHeight: 1.6,
                }}>
                  {getSupportActionSummary(openActions)}
                </div>
              </button>
            );
          })}
        </div>
      )}

      <div>
        <div style={{ fontSize: "10px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", marginBottom: 12 }}>
          Anchors · 8+ years
        </div>
        <div style={{ display: "flex", flexWrap: "wrap", gap: 8 }}>
          {anchors.map((emp) => (
            <button key={emp.id} onClick={() => onSelect(emp)} style={{
              background: "var(--bg-elevated)", border: "1px solid #D4A84344",
              padding: "6px 12px", cursor: "pointer",
              fontFamily: "'JetBrains Mono', monospace", fontSize: "10px", color: "#D4A843",
            }}>
              {emp.nickname ?? emp.full_name_en?.split(" ")[0] ?? "—"}
              <span style={{ color: "var(--text-muted)", marginLeft: 6 }}>{emp.tenure_years}yr</span>
            </button>
          ))}
        </div>
      </div>
    </div>
  );
}

// ─── Teams Screen with KPI panels ─────────────────────────────────────────────

function TeamsScreen({
  employees,
  projects,
  teams,
  kpis,
  onSelect,
  onOpenProject,
}: {
  employees: Employee[];
  projects: Project[];
  teams: TeamComposition[];
  kpis: DeptKpi[];
  onSelect: (emp: Employee) => void;
  onOpenProject: (projectCode: string) => void;
}) {
  const byDept: Record<string, Employee[]> = {};
  for (const emp of employees) {
    const key = emp.dept_code ?? "UNKNOWN";
    if (!byDept[key]) byDept[key] = [];
    byDept[key].push(emp);
  }
  const depts = Object.entries(byDept).sort((a, b) => b[1].length - a[1].length);

  const kpisByDept = useMemo(() => {
    const map: Record<string, DeptKpi[]> = {};
    for (const k of kpis) {
      if (!map[k.dept_code]) map[k.dept_code] = [];
      map[k.dept_code].push(k);
    }
    return map;
  }, [kpis]);

  const teamsByProject = useMemo(() => {
    const map = new Map<string, TeamComposition>();
    for (const team of teams) {
      map.set(team.project_code, team);
    }
    return map;
  }, [teams]);

  const employeeById = useMemo(
    () => new Map(employees.map((e) => [e.id, e])),
    [employees],
  );

  return (
    <div style={{ padding: "24px", maxWidth: "1200px", margin: "0 auto" }}>
      <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fill, minmax(300px, 1fr))", gap: 12 }}>
        {depts.map(([deptCode, members]) => {
          const color = DEPT_COLORS[deptCode] ?? "#666";
          const atRiskCount = members.filter((m) => ["risk", "watch"].includes(getRiskSignal(m))).length;
          const anchors = members.filter((m) => getRiskSignal(m) === "anchor").length;
          const deptKpis = kpisByDept[deptCode] ?? [];
          const deptProjects = projects
            .filter((project) => project.dept_code === deptCode)
            .sort((left, right) => getPriorityRank(left.priority) - getPriorityRank(right.priority))
            .slice(0, 3);
          return (
            <div key={deptCode} style={{
              background: "var(--bg-elevated)", border: `1px solid ${color}44`,
              borderTop: `3px solid ${color}`, padding: "12px 14px",
            }}>
              <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 8 }}>
                <span style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "11px", fontWeight: 700, color }}>{deptCode}</span>
                <span style={{ fontSize: "10px", color: "var(--text-muted)" }}>{members.length} people</span>
              </div>
              <div style={{ fontSize: "9px", color: "var(--text-secondary)", marginBottom: 10 }}>
                {members[0]?.dept_name_en ?? deptCode}
              </div>
              <div style={{ display: "flex", height: 4, gap: 1, marginBottom: 8 }}>
                {members.slice(0, 30).map((m) => {
                  const r = getRiskSignal(m);
                  return <div key={m.id} style={{ flex: 1, background: RISK_COLOR[r] }} />;
                })}
              </div>
              <div style={{ display: "flex", gap: 10, fontSize: "9px", marginBottom: 8 }}>
                {atRiskCount > 0 && <span style={{ color: "#C44D3F" }}>⚠ {atRiskCount} at risk</span>}
                {anchors > 0 && <span style={{ color: "#D4A843" }}>⚓ {anchors} anchors</span>}
              </div>

              {/* KPI strip */}
              {deptKpis.length > 0 && (
                <div style={{ marginBottom: 10, borderTop: "1px solid var(--border-subtle)", paddingTop: 8 }}>
                  <div style={{ fontSize: 8, color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase", marginBottom: 6 }}>
                    FY2025 KPIs · {deptKpis.length}
                  </div>
                  <div style={{ display: "flex", flexDirection: "column", gap: 3 }}>
                    {deptKpis.slice(0, 4).map((k) => {
                      const tgt = k.target_value ?? 0;
                      const act = k.actual_value ?? 0;
                      const pct = tgt > 0 ? Math.min(100, (act / tgt) * 100) : 0;
                      const hit = act > 0 && act >= tgt;
                      return (
                        <div key={k.id}>
                          <div style={{ display: "flex", justifyContent: "space-between", fontSize: 8, fontFamily: "'JetBrains Mono', monospace", color: "var(--text-secondary)" }}>
                            <span style={{ overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap", maxWidth: 200 }}>{k.kpi_name_en}</span>
                            <span style={{ color: "var(--text-muted)" }}>
                              {act || "—"}/{tgt || "—"}{k.target_unit ? ` ${k.target_unit}` : ""}
                            </span>
                          </div>
                          <div style={{ height: 3, background: "var(--border-subtle)", marginTop: 2 }}>
                            <div style={{
                              height: "100%",
                              width: `${pct}%`,
                              background: hit ? "#5B8C4A" : act > 0 ? color : "#444",
                              transition: "width 0.3s",
                            }} />
                          </div>
                        </div>
                      );
                    })}
                    {deptKpis.length > 4 && (
                      <div style={{ fontSize: 8, color: "var(--text-muted)", marginTop: 2 }}>+{deptKpis.length - 4} more</div>
                    )}
                  </div>
                </div>
              )}

              {deptProjects.length > 0 && (
                <div style={{ marginBottom: 10, borderTop: "1px solid var(--border-subtle)", paddingTop: 8 }}>
                  <div style={{ fontSize: 8, color: "var(--text-muted)", letterSpacing: "0.1em", textTransform: "uppercase", marginBottom: 6 }}>
                    Active Quests · {deptProjects.length}
                  </div>
                  <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
                    {deptProjects.map((project) => {
                      const team = teamsByProject.get(project.code);
                      const action = getProjectAction(project, team);
                      const assigned = getTeamMemberCount(team);
                      const weights = normalizeWeights(project.priority_weights);
                      const allocPcts = team?.allocation_pcts ?? {};
                      const archetypeSpend: Record<Archetype, number> = {
                        captain: 0, tech: 0, sales: 0, ops: 0, scout: 0, fighter: 0, goofoff: 0,
                      };
                      if (team) {
                        const roster: string[] = [];
                        if (team.coach_id) roster.push(team.coach_id);
                        for (const pid of team.player_ids ?? []) roster.push(pid);
                        for (const pid of roster) {
                          const emp = employeeById.get(pid);
                          if (!emp) continue;
                          const archetype = getArchetype(emp);
                          const pct = allocPcts[pid] ?? 100;
                          archetypeSpend[archetype] += effectiveCost(getEmployeeTokenCost(emp), pct);
                        }
                      }
                      const hasTeam = Boolean(team);
                      return (
                        <div key={project.code} style={{
                          border: `1px solid ${action.tone}44`,
                          background: "rgba(0,0,0,0.08)",
                          padding: "8px 9px",
                        }}>
                          <div style={{ display: "flex", justifyContent: "space-between", gap: 8 }}>
                            <div>
                              <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "9px", fontWeight: 700, color: "var(--text-primary)" }}>
                                {project.code} · {project.name}
                              </div>
                              <div style={{ fontSize: "8px", color: "var(--text-muted)", marginTop: 3 }}>
                                {project.priority} · {project.progress_pct}% progress · {assigned}/{project.team_size} staffed
                              </div>
                            </div>
                            <button onClick={() => onOpenProject(project.code)} style={{
                              border: "1px solid var(--border-subtle)",
                              background: "transparent",
                              color: "var(--text-secondary)",
                              cursor: "pointer",
                              fontFamily: "'JetBrains Mono', monospace",
                              fontSize: "8px",
                              padding: "4px 6px",
                              height: "fit-content",
                            }}>
                              Plan
                            </button>
                          </div>
                          <div style={{ display: "flex", gap: 4, marginTop: 8 }}>
                            {ARCHETYPES.map((archetype) => {
                              const weight = weights[archetype];
                              const spent = Math.round(archetypeSpend[archetype] * 10) / 10;
                              const fillPct = weight > 0
                                ? Math.min(140, (spent / weight) * 100)
                                : spent > 0 ? 140 : 0;
                              const over = weight > 0 && spent > weight + 0.05;
                              const tone = ARCHETYPE_COLOR[archetype];
                              const fillColor = over ? "#C44D3F" : tone;
                              return (
                                <div
                                  key={archetype}
                                  title={`${ARCHETYPE_LABEL[archetype]} · ${spent.toFixed(1)}/${weight} ⚡`}
                                  style={{ flex: 1, display: "flex", flexDirection: "column", gap: 2 }}
                                >
                                  <div style={{
                                    position: "relative",
                                    height: 6,
                                    background: `${tone}22`,
                                    borderRadius: 1,
                                    overflow: "hidden",
                                  }}>
                                    <div style={{
                                      position: "absolute",
                                      inset: 0,
                                      width: `${Math.min(100, fillPct)}%`,
                                      background: fillColor,
                                      transition: "width 0.25s",
                                    }} />
                                    {over && (
                                      <div style={{
                                        position: "absolute",
                                        top: 0,
                                        right: 0,
                                        bottom: 0,
                                        width: 2,
                                        background: "#C44D3F",
                                      }} />
                                    )}
                                  </div>
                                  <div style={{
                                    fontFamily: "'JetBrains Mono', monospace",
                                    fontSize: 7,
                                    letterSpacing: "0.04em",
                                    color: over ? "#C44D3F" : "var(--text-muted)",
                                    display: "flex",
                                    justifyContent: "space-between",
                                    gap: 2,
                                  }}>
                                    <span style={{ color: tone }}>
                                      {ARCHETYPE_LABEL[archetype].slice(0, 3).toUpperCase()}
                                    </span>
                                    <span>
                                      {hasTeam ? spent.toFixed(spent % 1 === 0 ? 0 : 1) : "·"}/{weight}
                                    </span>
                                  </div>
                                </div>
                              );
                            })}
                          </div>
                          {team && (
                            <div style={{ display: "flex", gap: 10, marginTop: 6, fontSize: "8px", color: "var(--text-secondary)" }}>
                              <span>Fit {team.fit_pct ?? "—"}%</span>
                              <span>Chem {team.chemistry_score ?? "—"}%</span>
                              <span>Overall {team.overall_score ?? "—"}%</span>
                            </div>
                          )}
                        </div>
                      );
                    })}
                  </div>
                </div>
              )}

              <div style={{ marginTop: 4, display: "flex", flexWrap: "wrap", gap: 4 }}>
                {members.slice(0, 8).map((m) => (
                  <button key={m.id} onClick={() => onSelect(m)} style={{
                    fontSize: "9px", padding: "2px 6px", background: "transparent",
                    border: "1px solid var(--border-subtle)", color: "var(--text-secondary)",
                    cursor: "pointer", fontFamily: "'JetBrains Mono', monospace",
                  }}>{m.nickname ?? m.full_name_en?.split(" ")[0] ?? "—"}</button>
                ))}
                {members.length > 8 && (
                  <span style={{ fontSize: "9px", color: "var(--text-muted)", padding: "2px 4px" }}>+{members.length - 8}</span>
                )}
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
}

  const [supportActions, setSupportActions] = useState<SupportActionRecord[]>([]);
  const [kpis, setKpis] = useState<DeptKpi[]>([]);
  const [loading, setLoading] = useState(true);
  const [screen, setScreen] = useState<Screen>("cockpit");
  const [search, setSearch] = useState("");
  const [deptFilter, setDeptFilter] = useState<string | null>(null);
  const [roleFilter, setRoleFilter] = useState<string | null>(null);
  const [selected, setSelected] = useState<Employee | null>(null);
  const [selectedProjectCode, setSelectedProjectCode] = useState<string | null>(null);
  const [now, setNow] = useState(() => new Date());
  const [activeRater, setActiveRater] = useState<RaterType>(getInitialRater);
  const searchRef = useRef<HTMLInputElement>(null);
  const deferredSearch = useDeferredValue(search);

  useEffect(() => {
    try { localStorage.setItem(ACTIVE_RATER_STORAGE_KEY, activeRater); } catch {}
  }, [activeRater]);

  useEffect(() => {
    let cancelled = false;

    async function load() {
      try {
        const [dashboardResponse, kpiResponse] = await Promise.all([
          fetch("/api/db/dashboard"),
          fetch("/api/db/kpis?cycle=FY2025"),
        ]);

        const dashboardData = await dashboardResponse.json();
        const kpiData = await kpiResponse.json();

        if (cancelled) return;

        setEmployees(Array.isArray(dashboardData.employees) ? dashboardData.employees : []);
        setProjects(Array.isArray(dashboardData.projects) ? dashboardData.projects : []);
        setTeams(Array.isArray(dashboardData.teams) ? dashboardData.teams : []);
        setSupportActions(
          Array.isArray(dashboardData.support_actions)
            ? sortSupportActions(dashboardData.support_actions.map(normalizeSupportAction))
            : [],
        );
        setKpis(Array.isArray(kpiData.kpis) ? kpiData.kpis : []);
      } catch {
        if (cancelled) return;
        setEmployees([]);
        setProjects([]);
        setTeams([]);
        setSupportActions([]);
        setKpis([]);
      } finally {
        if (!cancelled) setLoading(false);
      }
    }

    void load();

    return () => {
      cancelled = true;
    };
  }, []);

  useEffect(() => {
    const t = setInterval(() => setNow(new Date()), 30_000);
    return () => clearInterval(t);
  }, []);

  const allDepts = [...new Set(employees.map((e) => e.dept_code).filter(Boolean) as string[])].sort();
  const allRoles = [...new Set(employees.map((e) => e.role_level))].sort();

  const visible = employees.filter((emp) => {
    if (deptFilter && emp.dept_code !== deptFilter) return false;
    if (roleFilter && emp.role_level !== roleFilter) return false;
    if (deferredSearch) {
      const q = deferredSearch.toLowerCase();
      const name = `${emp.nickname ?? ""} ${emp.full_name_en ?? ""} ${emp.full_name_th}`.toLowerCase();
      if (!name.includes(q) && !(emp.dept_code ?? "").toLowerCase().includes(q)) return false;
    }
    return true;
  });

  const riskCounts = employees.reduce((acc, e) => {
    const r = getRiskSignal(e);
    acc[r] = (acc[r] ?? 0) + 1;
    return acc;
  }, {} as Record<string, number>);

  const selectedProject = selectedProjectCode
    ? projects.find((project) => project.code === selectedProjectCode) ?? null
    : null;
  const selectedTeam = selectedProject
    ? teams.find((team) => team.project_code === selectedProject.code)
    : undefined;

  function handleSaved(emp: Employee, updates: Partial<Employee>) {
    setEmployees((prev) => prev.map((e) => e.id === emp.id ? { ...e, ...updates } : e));
    if (selected?.id === emp.id) setSelected((s) => s ? { ...s, ...updates } : s);
  }

  function focusDepartment(deptCode: string) {
    setDeptFilter(deptCode);
    setScreen("teams");
  }

  function openProjectPlanner(projectCode: string) {
    setSelectedProjectCode(projectCode);
  }

  async function refreshSavedTeam(projectCode: string) {
    try {
      const response = await fetch(`/api/db/teams?projectCode=${encodeURIComponent(projectCode)}`);
      const data = await response.json();
      if (!response.ok) return;

      const nextTeam = Array.isArray(data.teams) ? data.teams[0] : null;
      setTeams((current) => {
        const withoutProject = current.filter((team) => team.project_code !== projectCode);
        return nextTeam ? [...withoutProject, nextTeam] : withoutProject;
      });
    } catch {
      return;
    }
  }

  function handleProjectPrioritySaved(projectCode: string, priorityWeights: PriorityWeights) {
    setProjects((current) => current.map((project) => (
      project.code === projectCode
        ? { ...project, priority_weights: priorityWeights }
        : project
    )));
  }

  return (
    <div className="tabletop" style={{ height: "100vh", display: "flex", flexDirection: "column", overflow: "hidden", color: "var(--text-primary)" }}>
      <header style={{
        display: "flex", alignItems: "center", justifyContent: "space-between",
        padding: "0 20px", height: "44px", flexShrink: 0,
        background: "rgba(18,12,6,0.95)", backdropFilter: "blur(4px)",
        borderBottom: "1px solid var(--border-subtle)",
      }}>
        <div style={{ display: "flex", alignItems: "center", gap: 20 }}>
          <span style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: "11px", fontWeight: 700, color: "var(--accent-gold)", letterSpacing: "0.1em" }}>
            TKC TALENT
          </span>
          <div style={{ display: "flex", gap: 2 }}>
            {(["cockpit", "roster", "teams", "signals"] as Screen[]).map((s) => (
              <button key={s} onClick={() => setScreen(s)} style={{
                padding: "4px 12px", fontSize: "9px", letterSpacing: "0.1em",
                textTransform: "uppercase", cursor: "pointer",
                fontFamily: "'JetBrains Mono', monospace",
                background: screen === s ? "rgba(212,168,67,0.12)" : "transparent",
                border: screen === s ? "1px solid var(--accent-gold)" : "1px solid transparent",
                color: screen === s ? "var(--accent-gold)" : "var(--text-muted)",
                transition: "all 0.15s",
              }}>{s === "cockpit" ? "Cockpit" : s === "roster" ? "Roster" : s === "teams" ? "Teams" : "Signals"}</button>
            ))}
          </div>
          <div style={{ width: 1, height: 16, background: "var(--border-subtle)" }} />
          <FiscalTicker employeeCount={employees.length} />
        </div>

        <div style={{ display: "flex", alignItems: "center", gap: 16, fontSize: "10px", fontFamily: "'JetBrains Mono', monospace" }}>
          <RoleSwitcher value={activeRater} onChange={setActiveRater} />
          <span style={{ color: "var(--text-muted)" }}>{employees.length} heroes</span>
          <span style={{ color: "var(--text-muted)" }}>{projects.length} quests</span>
          {riskCounts.risk > 0 && <span style={{ color: "#C44D3F" }}>⚠ {riskCounts.risk} watch</span>}
          {riskCounts.anchor > 0 && <span style={{ color: "#D4A843" }}>⚓ {riskCounts.anchor} anchors</span>}
          <span style={{ color: "var(--text-muted)" }}>
            {now.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" })}
          </span>
          <span style={{ color: "var(--text-secondary)" }}>
            {now.toLocaleTimeString("en-GB", { hour: "2-digit", minute: "2-digit", hour12: false })}
          </span>
        </div>
      </header>

      <div style={{ display: "flex", flex: 1, minHeight: 0 }}>
        <aside style={{
          width: "220px", flexShrink: 0,
          background: "rgba(18,12,6,0.92)", backdropFilter: "blur(4px)",
          borderRight: "1px solid var(--border-subtle)",
          display: "flex", flexDirection: "column", padding: "14px 12px", gap: 14,
          overflowY: "auto",
        }}>
          <input
            ref={searchRef} type="text" placeholder="Search name, dept..."
            value={search} onChange={(e) => setSearch(e.target.value)}
            style={{
              background: "var(--bg-elevated)", border: "1px solid var(--border-subtle)",
              color: "var(--text-primary)", fontSize: "11px", padding: "7px 10px",
              fontFamily: "'JetBrains Mono', monospace", outline: "none", width: "100%",
            }}
          />
          <div style={{ fontSize: "9px", color: "var(--text-muted)", letterSpacing: "0.08em" }}>
            {visible.length} / {employees.length} shown
          </div>

          <div>
            <div style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", marginBottom: 6 }}>Department</div>
            <div style={{ display: "flex", flexDirection: "column", gap: 2 }}>
              <FilterPill active={deptFilter === null} color="var(--accent-gold)" onClick={() => setDeptFilter(null)}>All</FilterPill>
              {allDepts.map((d) => (
                <FilterPill key={d} active={deptFilter === d} color={DEPT_COLORS[d] ?? "#666"} onClick={() => setDeptFilter(deptFilter === d ? null : d)}>{d}</FilterPill>
              ))}
            </div>
          </div>

          <div>
            <div style={{ fontSize: "8px", color: "var(--text-muted)", letterSpacing: "0.12em", textTransform: "uppercase", marginBottom: 6 }}>Role</div>
            <div style={{ display: "flex", flexDirection: "column", gap: 2 }}>
              <FilterPill active={roleFilter === null} color="var(--accent-gold)" onClick={() => setRoleFilter(null)}>All</FilterPill>
              {allRoles.map((r) => (
                <FilterPill key={r} active={roleFilter === r} color="var(--text-secondary)" onClick={() => setRoleFilter(roleFilter === r ? null : r)}>{ROLE_LABEL[r] ?? r}</FilterPill>
              ))}
            </div>
          </div>
        </aside>

        <main style={{ flex: 1, minWidth: 0, overflowY: "auto" }}>
          {loading && (
            <div style={{ display: "flex", alignItems: "center", justifyContent: "center", height: "100%", color: "var(--text-muted)", fontFamily: "'JetBrains Mono', monospace", fontSize: "12px" }}>
              Loading heroes...
            </div>
          )}

          {!loading && screen === "cockpit" && (
            <CockpitScreen
              employees={employees}
              projects={projects}
              teams={teams}
              supportActions={supportActions}
              kpis={kpis}
              onSelect={setSelected}
              onOpenScreen={setScreen}
              onFocusDept={focusDepartment}
              onOpenProject={openProjectPlanner}
            />
          )}

          {!loading && screen === "roster" && (
            <div style={{ padding: "20px 24px" }}>
              <div style={{ display: "flex", flexWrap: "wrap", gap: "16px" }}>
                {visible.map((emp) => (
                  <PolaroidCard key={emp.id} emp={emp} onClick={() => setSelected(emp)} />
                ))}
                {visible.length === 0 && (
                  <div style={{ color: "var(--text-muted)", fontSize: "12px", fontFamily: "'JetBrains Mono', monospace" }}>
                    No employees match the filter.
                  </div>
                )}
              </div>
            </div>
          )}

          {!loading && screen === "teams" && (
            <TeamsScreen
              employees={visible}
              projects={projects}
              teams={teams}
              kpis={kpis}
              onSelect={setSelected}
              onOpenProject={openProjectPlanner}
            />
          )}
          {!loading && screen === "signals" && (
            <SignalsScreen
              employees={employees}
              supportActions={supportActions}
              onSelect={setSelected}
            />
          )}
        </main>
      </div>

      {selected && (
        <EmployeeDrawer
          emp={selected}
          employees={employees}
          activeRater={activeRater}
          onClose={() => setSelected(null)}
          onSaved={(updates) => handleSaved(selected, updates)}
        />
      )}

      {selectedProject && (
        <ProjectStaffingDrawer
          project={selectedProject}
          team={selectedTeam}
          employees={employees}
          onClose={() => setSelectedProjectCode(null)}
          onSaved={(projectCode) => { void refreshSavedTeam(projectCode); }}
          onProjectUpdated={handleProjectPrioritySaved}
        />
      )}
    </div>
  );
}

function FilterPill({ active, color, onClick, children }: {
  active: boolean; color: string; onClick: () => void; children: React.ReactNode;
}) {
  return (
    <button onClick={onClick} style={{
      padding: "3px 8px", fontSize: "9px", cursor: "pointer", textAlign: "left",
      fontFamily: "'JetBrains Mono', monospace",
      background: active ? `${color}22` : "transparent",
      border: active ? `1px solid ${color}88` : "1px solid transparent",
      color: active ? color : "var(--text-muted)", transition: "all 0.12s",
    }}>{children}</button>
  );
}
